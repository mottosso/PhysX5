80template <class Listener, class Base>
81class PxBroadcast : public Base
82{
83 public:
84 static const uint32_t MAX_NB_LISTENERS = 16;
85
89 PxBroadcast()
90 {
91 }
92
101 void registerListener(Listener& listener)
102 {
103 if(mListeners.size() < MAX_NB_LISTENERS)
104 mListeners.pushBack(&listener);
105 }
106
115 void deregisterListener(Listener& listener)
116 {
117 mListeners.findAndReplaceWithLast(&listener);
118 }
119
125 uint32_t getNbListeners() const
126 {
127 return mListeners.size();
128 }
129
136 Listener& getListener(uint32_t index)
137 {
138 PX_ASSERT(index <= mListeners.size());
139 return *mListeners[index];
140 }
141
142 protected:
143 virtual ~PxBroadcast()
144 {
145 }
146
147 physx::PxInlineArray<Listener*, MAX_NB_LISTENERS, physx::PxAllocator> mListeners;
148};

154class PxBroadcastingAllocator : public PxBroadcast<PxAllocationListener, PxAllocatorCallback>
155{
156 PX_NOCOPY(PxBroadcastingAllocator)
157
158 public:
162 PxBroadcastingAllocator(PxAllocatorCallback& allocator, PxErrorCallback& error) : mAllocator(allocator), mError(error)
163 {
164 mListeners.clear();
165 }
166
170 virtual ~PxBroadcastingAllocator()
171 {
172 mListeners.clear();
173 }
174
190 void* allocate(size_t size, const char* typeName, const char* filename, int line)
191 {
192 void* mem = mAllocator.allocate(size, typeName, filename, line);
193
194 if(!mem)
195 {
196 mError.reportError(PxErrorCode::eABORT, "User allocator returned NULL.", __FILE__, __LINE__);
197 return NULL;
198 }
199
200 if((size_t(mem) & 15))
201 {
202 mError.reportError(PxErrorCode::eABORT, "Allocations must be 16-byte aligned.", __FILE__, __LINE__);
203 return NULL;
204 }
205
206 for(uint32_t i = 0; i < mListeners.size(); i++)
207 mListeners[i]->onAllocation(size, typeName, filename, line, mem);
208
209 return mem;
210 }
211
220 void deallocate(void* ptr)
221 {
222 for(uint32_t i = 0; i < mListeners.size(); i++)
223 {
224 mListeners[i]->onDeallocation(ptr);
225 }
226 mAllocator.deallocate(ptr);
227 }
228
229 private:
230 PxAllocatorCallback& mAllocator;
231 PxErrorCallback& mError;
232};
