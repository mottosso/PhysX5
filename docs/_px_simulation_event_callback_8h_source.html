175struct PxContactPairExtraDataIterator
176{
182 PX_FORCE_INLINE PxContactPairExtraDataIterator(const PxU8* stream, PxU32 size)
183 : currPtr(stream), endPtr(stream + size), contactPairIndex(0)
184 {
185 clearDataPtrs();
186 }
187
206 PX_INLINE bool nextItemSet()
207 {
208 clearDataPtrs();
209
210 bool foundEntry = false;
211 bool endOfItemSet = false;
212 while ((currPtr < endPtr) && (!endOfItemSet))
213 {
214 const PxContactPairExtraDataItem* edItem = reinterpret_cast<const PxContactPairExtraDataItem*>(currPtr);
215 PxU8 type = edItem->type;
216
217 switch(type)
218 {
219 case PxContactPairExtraDataType::ePRE_SOLVER_VELOCITY:
220 {
221 PX_ASSERT(!preSolverVelocity);
222 preSolverVelocity = static_cast<const PxContactPairVelocity*>(edItem);
223 currPtr += sizeof(PxContactPairVelocity);
224 foundEntry = true;
225 }
226 break;
227
228 case PxContactPairExtraDataType::ePOST_SOLVER_VELOCITY:
229 {
230 postSolverVelocity = static_cast<const PxContactPairVelocity*>(edItem);
231 currPtr += sizeof(PxContactPairVelocity);
232 foundEntry = true;
233 }
234 break;
235
236 case PxContactPairExtraDataType::eCONTACT_EVENT_POSE:
237 {
238 eventPose = static_cast<const PxContactPairPose*>(edItem);
239 currPtr += sizeof(PxContactPairPose);
240 foundEntry = true;
241 }
242 break;
243
244 case PxContactPairExtraDataType::eCONTACT_PAIR_INDEX:
245 {
246 if (!foundEntry)
247 {
248 contactPairIndex = static_cast<const PxContactPairIndex*>(edItem)->index;
249 currPtr += sizeof(PxContactPairIndex);
250 foundEntry = true;
251 }
252 else
253 endOfItemSet = true;
254 }
255 break;
256
257 default:
258 return foundEntry;
259 }
260 }
261
262 return foundEntry;
263 }
264
265private:
269 PX_FORCE_INLINE void clearDataPtrs()
270 {
271 preSolverVelocity = NULL;
272 postSolverVelocity = NULL;
273 eventPose = NULL;
274 }
275
276public:
280 const PxU8* currPtr;
281
285 const PxU8* endPtr;
286
292 const PxContactPairVelocity* preSolverVelocity;
293
299 const PxContactPairVelocity* postSolverVelocity;
300
306 const PxContactPairPose* eventPose;
307
313 PxU32 contactPairIndex;
314};

626PX_INLINE PxU32 PxContactPair::extractContacts(PxContactPairPoint* userBuffer, PxU32 bufferSize) const
627{
628 PxU32 nbContacts = 0;
629
630 if(contactCount && bufferSize)
631 {
632 PxContactStreamIterator iter(contactPatches, contactPoints, getInternalFaceIndices(), patchCount, contactCount);
633
634 const PxReal* impulses = contactImpulses;
635
636 const PxU32 flippedContacts = (flags & PxContactPairFlag::eINTERNAL_CONTACTS_ARE_FLIPPED);
637 const PxU32 hasImpulses = (flags & PxContactPairFlag::eINTERNAL_HAS_IMPULSES);
638
639 while(iter.hasNextPatch())
640 {
641 iter.nextPatch();
642 while(iter.hasNextContact())
643 {
644 iter.nextContact();
645 PxContactPairPoint& dst = userBuffer[nbContacts];
646 dst.position = iter.getContactPoint();
647 dst.separation = iter.getSeparation();
648 dst.normal = iter.getContactNormal();
649 if(!flippedContacts)
650 {
651 dst.internalFaceIndex0 = iter.getFaceIndex0();
652 dst.internalFaceIndex1 = iter.getFaceIndex1();
653 }
654 else
655 {
656 dst.internalFaceIndex0 = iter.getFaceIndex1();
657 dst.internalFaceIndex1 = iter.getFaceIndex0();
658 }
659
660 if(hasImpulses)
661 {
662 const PxReal impulse = impulses[nbContacts];
663 dst.impulse = dst.normal * impulse;
664 }
665 else
666 dst.impulse = PxVec3(0.0f);
667 ++nbContacts;
668 if(nbContacts == bufferSize)
669 return nbContacts;
670 }
671 }
672 }
673
674 return nbContacts;
675}

678PX_INLINE void PxContactPair::bufferContacts(PxContactPair* newPair, PxU8* bufferMemory) const
679{
680 PxU8* patches = bufferMemory;
681 PxU8* contacts = NULL;
682 if(patches)
683 {
684 contacts = bufferMemory + patchCount * sizeof(PxContactPatch);
685 PxMemCopy(patches, contactPatches, sizeof(PxContactPatch)*patchCount);
686 PxMemCopy(contacts, contactPoints, contactStreamSize - (sizeof(PxContactPatch)*patchCount));
687 }
688
689 if(contactImpulses)
690 {
691 PxMemCopy(bufferMemory + ((contactStreamSize + 15) & (~15)), contactImpulses, sizeof(PxReal) * contactCount);
692 }
693
694 if (newPair)
695 {
696 *newPair = *this;
697 newPair->contactPatches = patches;
698 newPair->contactPoints = contacts;
699 }
700}
