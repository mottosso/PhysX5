113 void doAddName( const char* inName, uint16_t inEventId, bool inCompileTimeEnabled )
114 {
115 TLockType theLocker( mMutex );
116 mEvtIdToNameMap.insert( inEventId, inName );
117 uint32_t idx = static_cast<uint32_t>( mEventNames.size() );
118 mNameToEvtIndexMapRW.insert( inName, idx );
119 mEventNames.pushBack( PxProfileEventName( inName, PxProfileEventId( inEventId, inCompileTimeEnabled ) ) );
120 }

122 virtual void flushEventIdNameMap()
123 {
124 // copy the RW map into R map
125 if (mNameToEvtIndexMapRW.size())
126 {
127 for (TNameToEvtIndexMap::Iterator iter = mNameToEvtIndexMapRW.getIterator(); !iter.done(); ++iter)
128 {
129 mNameToEvtIndexMapR.insert(iter->first, iter->second);
130 }
131 mNameToEvtIndexMapRW.clear();
132 }

140 virtual uint16_t getEventIdsForNames( const char** inNames, uint32_t inLen )
141 {
142 if ( inLen == 0 )
143 return 0;
144
145 // search the read-only map first
146 const TNameToEvtIndexMap::Entry* theEntry( mNameToEvtIndexMapR.find( inNames[0] ) );
147 if ( theEntry )
148 return mEventNames[theEntry->second].eventId;
149
150 TLockType theLocker(mMutex);
151
152 const TNameToEvtIndexMap::Entry* theReEntry(mNameToEvtIndexMapRW.find(inNames[0]));
153 if (theReEntry)
154 return mEventNames[theReEntry->second].eventId;
155
156 //Else git R dun.
157 uint16_t nameSize = static_cast<uint16_t>( mEventNames.size() );
158 //We don't allow 0 as an event id.
159 uint16_t eventId = nameSize;
160 //Find a contiguous set of unique event ids
161 bool foundAnEventId = false;
162 do
163 {
164 foundAnEventId = false;
165 ++eventId;
166 for ( uint16_t idx = 0; idx < inLen && foundAnEventId == false; ++idx )
167 foundAnEventId = mEvtIdToNameMap.find( uint16_t(eventId + idx) ) != NULL;
168 }
169 while( foundAnEventId );
170
171 uint32_t clientCount = mZoneClients.size();
172 for ( uint16_t nameIdx = 0; nameIdx < inLen; ++nameIdx )
173 {
174 uint16_t newId = uint16_t(eventId + nameIdx);
175 doAddName( inNames[nameIdx], newId, true );
176 for( uint32_t clientIdx =0; clientIdx < clientCount; ++clientIdx )
177 mZoneClients[clientIdx]->handleEventAdded( PxProfileEventName( inNames[nameIdx], PxProfileEventId( newId ) ) );
178 }
179
180 return eventId;
181 }

209 {
210 TLockType lock( mMutex );
211 for ( uint32_t idx =0; idx < mZoneClients.size(); ++idx )
212 {
213 if (mZoneClients[idx] == &inClient )

215 inClient.handleClientRemoved();
