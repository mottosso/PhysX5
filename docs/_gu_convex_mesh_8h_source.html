50 PX_INLINE PxU32 computeBufferSize(const Gu::ConvexHullData& data, PxU32 nb)
51 {
52 PxU32 bytesNeeded = sizeof(Gu::HullPolygonData) * data.mNbPolygons;
53 bytesNeeded += sizeof(PxVec3) * data.mNbHullVertices;
54 bytesNeeded += sizeof(PxU8) * data.mNbEdges * 2; // mFacesByEdges8
55 bytesNeeded += sizeof(PxU8) * data.mNbHullVertices * 3; // mFacesByVertices8;
56 bytesNeeded += data.mNbEdges.isBitSet() ? (sizeof(PxU16) * data.mNbEdges * 2) : 0; // mEdges;
57 bytesNeeded += sizeof(PxU8) * nb; // mVertexData8
58
59 //4 align the whole thing!
60 const PxU32 mod = bytesNeeded % sizeof(PxReal);
61 if (mod)
62 bytesNeeded += sizeof(PxReal) - mod;
63 return bytesNeeded;
64 }
