47 class ZoneManagerImpl : public PxProfileZoneManager
48 {
49 typedef ScopedLockImpl<PxMutex> TScopedLockType;
50 PxProfileAllocatorWrapper mWrapper;
51 PxProfileArray<PxProfileZone*> mZones;
52 PxProfileArray<PxProfileZoneHandler*> mHandlers;
53 PxMutex mMutex;

55 ZoneManagerImpl( const ZoneManagerImpl& inOther );
56 ZoneManagerImpl& operator=( const ZoneManagerImpl& inOther );

58 public:
59
60 ZoneManagerImpl(PxAllocatorCallback* inFoundation)
61 : mWrapper( inFoundation )
62 , mZones( mWrapper )
63 , mHandlers( mWrapper )
64 {}
65
66 virtual ~ZoneManagerImpl()
67 {
68 //This assert would mean that a profile zone is outliving us.
69 //This will cause a crash when the profile zone is released.
70 PX_ASSERT( mZones.size() == 0 );
71 while( mZones.size() )
72 removeProfileZone( *mZones.back() );
73 }
74
75 virtual void addProfileZone( PxProfileZone& inSDK )
76 {
77 TScopedLockType lock( &mMutex );
78
79 if ( inSDK.getProfileZoneManager() != NULL )
80 {
81 if ( inSDK.getProfileZoneManager() == this )
82 return;
83 else //there must be two managers in the system somehow.
84 {
85 PX_ASSERT( false );
86 inSDK.getProfileZoneManager()->removeProfileZone( inSDK );
87 }
88 }
89 mZones.pushBack( &inSDK );
90 inSDK.setProfileZoneManager( this );
91 for ( uint32_t idx =0; idx < mHandlers.size(); ++idx )
92 mHandlers[idx]->onZoneAdded( inSDK );
93 }
94
95 virtual void removeProfileZone( PxProfileZone& inSDK )
96 {
97 TScopedLockType lock( &mMutex );
98 if ( inSDK.getProfileZoneManager() == NULL )
99 return;
100
101 else if ( inSDK.getProfileZoneManager() != this )
102 {
103 PX_ASSERT( false );
104 inSDK.getProfileZoneManager()->removeProfileZone( inSDK );
105 return;
106 }
107
108 inSDK.setProfileZoneManager( NULL );
109 for ( uint32_t idx = 0; idx < mZones.size(); ++idx )
110 {
111 if ( mZones[idx] == &inSDK )
112 {
113 for ( uint32_t handler =0; handler < mHandlers.size(); ++handler )
114 mHandlers[handler]->onZoneRemoved( inSDK );
115 mZones.replaceWithLast( idx );
116 }
117 }
118 }
119
120 virtual void flushProfileEvents()
121 {
122 uint32_t sdkCount = mZones.size();
123 for ( uint32_t idx = 0; idx < sdkCount; ++idx )
124 mZones[idx]->flushProfileEvents();
125 }
126
127 virtual void addProfileZoneHandler( PxProfileZoneHandler& inHandler )
128 {
129 TScopedLockType lock( &mMutex );
130 mHandlers.pushBack( &inHandler );
131 for ( uint32_t idx = 0; idx < mZones.size(); ++idx )
132 inHandler.onZoneAdded( *mZones[idx] );
133 }
134
135 virtual void removeProfileZoneHandler( PxProfileZoneHandler& inHandler )
136 {
137 TScopedLockType lock( &mMutex );
138 for( uint32_t idx = 0; idx < mZones.size(); ++idx )
139 inHandler.onZoneRemoved( *mZones[idx] );
140 for( uint32_t idx = 0; idx < mHandlers.size(); ++idx )
141 {
142 if ( mHandlers[idx] == &inHandler )
143 mHandlers.replaceWithLast( idx );
144 }
145 }
146
147 virtual PxProfileZone& createProfileZone( const char* inSDKName, PxProfileNameProvider* inProvider, uint32_t inEventBufferByteSize )
148 {
149 NullEventNameProvider nullProvider;
150 if ( inProvider == NULL )
151 inProvider = &nullProvider;
152 return createProfileZone( inSDKName, inProvider->getProfileNames(), inEventBufferByteSize );
153 }
154
155
156 virtual PxProfileZone& createProfileZone( const char* inSDKName, PxProfileNames inNames, uint32_t inEventBufferByteSize )
157 {
158 PxProfileZone& retval( PxProfileZone::createProfileZone( &mWrapper.getAllocator(), inSDKName, inNames, inEventBufferByteSize ) );
159 addProfileZone( retval );
160 return retval;
161 }
162
163 virtual void release()
164 {
165 PX_PROFILE_DELETE( mWrapper.getAllocator(), this );
166 }
167 };
168} }
