156 PX_FORCE_INLINE aos::Vec4V getProjection(const aos::Vec3VArg dir) const
157 {
158 using namespace aos;
159
160 const Vec4V dx = V4Scale(vertsX, V3GetX(dir));
161 const Vec4V dy = V4Scale(vertsY, V3GetY(dir));
162 const Vec4V dz = V4Scale(vertsZ, V3GetZ(dir));
163
164 return V4Add(dx, V4Add(dy, dz));
