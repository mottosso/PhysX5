53
54 PX_FORCE_INLINE bool isValid(const PxVehicleAxleDescription& axleDesc) const
55 {
56 if (!autoboxParams.isValid(gearBoxParams))
57 return false;
58 if (!clutchCommandResponseParams.isValid())
59 return false;
60 if (!engineParams.isValid())
61 return false;
62 if (!gearBoxParams.isValid())
63 return false;
64 if (!multiWheelDifferentialParams.isValid(axleDesc))
65 return false;
66 if (!fourWheelDifferentialParams.isValid(axleDesc))
67 return false;
68 if (!tankDifferentialParams.isValid(axleDesc))
69 return false;
70 if (!clutchParams.isValid())
71 return false;
72 return true;
73 }
74};

181 virtual void getDataForEngineDriveCommandResponseComponent(
182 const PxVehicleAxleDescription*& axleDescription,
183 PxVehicleSizedArrayData<const PxVehicleBrakeCommandResponseParams>& brakeResponseParams,
184 const PxVehicleSteerCommandResponseParams*& steerResponseParams,
185 PxVehicleSizedArrayData<const PxVehicleAckermannParams>& ackermannParams,
186 const PxVehicleGearboxParams*& gearboxParams,
187 const PxVehicleClutchCommandResponseParams*& clutchResponseParams,
188 const PxVehicleEngineParams*& engineParams,
189 const PxVehicleRigidBodyState*& rigidBodyState,
190 const PxVehicleEngineState*& engineState,
191 const PxVehicleAutoboxParams*& autoboxParams,
192 const PxVehicleCommandState*& commands,
193 const PxVehicleEngineDriveTransmissionCommandState*& transmissionCommands,
194 PxVehicleArrayData<PxReal>& brakeResponseStates,
195 PxVehicleEngineDriveThrottleCommandResponseState*& throttleResponseState,
196 PxVehicleArrayData<PxReal>& steerResponseStates,
197 PxVehicleGearboxState*& gearboxResponseState,
198 PxVehicleClutchCommandResponseState*& clutchResponseState,
199 PxVehicleAutoboxState*& autoboxState)
200 {
201 axleDescription = &mBaseParams.axleDescription;
202 brakeResponseParams.setDataAndCount(mBaseParams.brakeResponseParams, sizeof(mBaseParams.brakeResponseParams) / sizeof(PxVehicleBrakeCommandResponseParams));
203 steerResponseParams = &mBaseParams.steerResponseParams;
204 ackermannParams.setDataAndCount(mBaseParams.ackermannParams, sizeof(mBaseParams.ackermannParams)/sizeof(PxVehicleAckermannParams));
205 gearboxParams = &mEngineDriveParams.gearBoxParams;
206 clutchResponseParams = &mEngineDriveParams.clutchCommandResponseParams;
207 engineParams = &mEngineDriveParams.engineParams;
208 rigidBodyState = &mBaseState.rigidBodyState;
209 engineState = &mEngineDriveState.engineState;
210 autoboxParams = &mEngineDriveParams.autoboxParams;
211 commands = &mCommandState;
212 transmissionCommands = (Enum::eDIFFTYPE_TANKDRIVE == mDifferentialType) ? &mTankDriveTransmissionCommandState : &mTransmissionCommandState;
213 brakeResponseStates.setData(mBaseState.brakeCommandResponseStates);
214 throttleResponseState = &mEngineDriveState.throttleCommandResponseState;
215 steerResponseStates.setData(mBaseState.steerCommandResponseStates);
216 gearboxResponseState = &mEngineDriveState.gearboxState;
217 clutchResponseState = &mEngineDriveState.clutchCommandResponseState;
218 autoboxState = &mEngineDriveState.autoboxState;
219 }

312 gearboxState = &mEngineDriveState.gearboxState;
313 clutchState = &mEngineDriveState.clutchState;
314 }
