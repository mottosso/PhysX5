41class PxcScratchAllocator : public PxUserAllocated
42{
43 PX_NOCOPY(PxcScratchAllocator)
44public:
45 PxcScratchAllocator() : mStack("PxcScratchAllocator"), mStart(NULL), mSize(0)
46 {
47 mStack.reserve(64);
48 mStack.pushBack(0);
49 }
50
51 void setBlock(void* addr, PxU32 size)
52 {
53 PX_ASSERT(!(size&15));
54
55 // if the stack is not empty then some scratch memory was not freed on the previous frame. That's
56 // likely indicative of a problem, because when the scratch block is too small the memory will have
57 // come from the heap
58
59 PX_ASSERT(mStack.size()==1);
60 mStack.popBack();
61
62 mStart = reinterpret_cast<PxU8*>(addr);
63 mSize = size;
64 mStack.pushBack(mStart + size);
65 }
66
67 void* allocAll(PxU32& size)
68 {
69 PxMutex::ScopedLock lock(mLock);
70 PX_ASSERT(mStack.size()>0);
71 size = PxU32(mStack.back()-mStart);
72
73 if(size==0)
74 return NULL;
75
76 mStack.pushBack(mStart);
77 return mStart;
78 }
79
80 void* alloc(PxU32 requestedSize, bool fallBackToHeap = false)
81 {
82 requestedSize = (requestedSize+15)&~15;
83
84 PxMutex::ScopedLock lock(mLock);
85 PX_ASSERT(mStack.size()>=1);
86
87 PxU8* top = mStack.back();
88
89 if(top - mStart >= ptrdiff_t(requestedSize))
90 {
91 PxU8* addr = top - requestedSize;
92 mStack.pushBack(addr);
93 return addr;
94 }
95
96 if(!fallBackToHeap)
97 return NULL;
98
99 return PX_ALLOC(requestedSize, "Scratch Block Fallback");
100 }
101
102 void free(void* addr)
103 {
104 PX_ASSERT(addr!=NULL);
105 if(!isScratchAddr(addr))
106 {
107 PX_FREE(addr);
108 return;
109 }
110
111 PxMutex::ScopedLock lock(mLock);
112 PX_ASSERT(mStack.size()>1);
113
114 PxU32 i=mStack.size()-1;
115 while(mStack[i]<addr)
116 i--;
117
118 PX_ASSERT(mStack[i]==addr);
119 mStack.remove(i);
120 }
121
122 bool isScratchAddr(void* addr) const
123 {
124 PxU8* a = reinterpret_cast<PxU8*>(addr);
125 return a>= mStart && a<mStart+mSize;
126 }
127
128private:
129 PxMutex mLock;
130 PxArray<PxU8*> mStack;
131 PxU8* mStart;
132 PxU32 mSize;
133};
