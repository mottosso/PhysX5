113 PX_CUDA_CALLABLE PX_FORCE_INLINE bool isIdentity() const
114 {
115 return (scale.x == 1.0f && scale.y == 1.0f && scale.z == 1.0f);
116 }
117
121 PX_CUDA_CALLABLE PX_FORCE_INLINE PxMeshScale getInverse() const
122 {
123 return PxMeshScale(PxVec3(1.0f/scale.x, 1.0f/scale.y, 1.0f/scale.z), rotation);
124 }
125
129 PX_CUDA_CALLABLE PX_FORCE_INLINE PxMat33 toMat33() const
130 {
131 PxMat33 rot(rotation);
132 PxMat33 trans = rot.getTranspose();
133 trans.column0 *= scale[0];
134 trans.column1 *= scale[1];
135 trans.column2 *= scale[2];
136 return trans * rot;
137 }
138
142 PX_CUDA_CALLABLE PX_FORCE_INLINE bool hasNegativeDeterminant() const
143 {
144 return (scale.x * scale.y * scale.z < 0.0f);
145 }
146
147 PxVec3 transform(const PxVec3& v) const
148 {
149 return rotation.rotateInv(scale.multiply(rotation.rotate(v)));
150 }
151
152 bool isValidForTriangleMesh() const
153 {
154 PxVec3 absXYZ = scale.abs();
155 return (absXYZ.maxElement() <= PX_MESH_SCALE_MAX) && (absXYZ.minElement() >= PX_MESH_SCALE_MIN);
156 }
157
158 bool isValidForConvexMesh() const
159 {
160 return (scale.maxElement() <= PX_MESH_SCALE_MAX) && (scale.minElement() >= PX_MESH_SCALE_MIN);
161 }
162
163 PxVec3 scale;
164 PxQuat rotation;
165};
