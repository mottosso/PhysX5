45 template<typename TMutex,
46 typename TScopedLock>
47 class MemoryEventBuffer : public DataBuffer<TMutex, TScopedLock>
48 {
49 public:
50 typedef DataBuffer<TMutex, TScopedLock> TBaseType;
51 typedef typename TBaseType::TMutexType TMutexType;
52 typedef typename TBaseType::TScopedLockType TScopedLockType;
53 typedef typename TBaseType::TU8AllocatorType TU8AllocatorType;
54 typedef typename TBaseType::TMemoryBufferType TMemoryBufferType;
55 typedef typename TBaseType::TBufferClientArray TBufferClientArray;
56 typedef PxHashMap<const char*, uint32_t, PxHash<const char*>, TU8AllocatorType> TCharPtrToHandleMap;
57
58 protected:
59 TCharPtrToHandleMap mStringTable;
60
61 public:
62
63 MemoryEventBuffer( PxAllocatorCallback& cback
64 , uint32_t inBufferFullAmount
65 , TMutexType* inBufferMutex )
66 : TBaseType( &cback, inBufferFullAmount, inBufferMutex, "struct physx::profile::MemoryEvent" )
67 , mStringTable( TU8AllocatorType( TBaseType::getWrapper(), "MemoryEventStringBuffer" ) )
68 {
69 }
70
71 uint32_t getHandle( const char* inData )
72 {
73 if ( inData == NULL ) inData = "";
74 const typename TCharPtrToHandleMap::Entry* result( mStringTable.find( inData ) );
75 if ( result )
76 return result->second;
77 uint32_t hdl = mStringTable.size() + 1;
78 mStringTable.insert( inData, hdl );
79 StringTableEvent theEvent;
80 theEvent.init( inData, hdl );
81 sendEvent( theEvent );
82 return hdl;
83 }
84
85 void onAllocation( size_t inSize, const char* inType, const char* inFile, uint32_t inLine, uint64_t addr )
86 {
87 if ( addr == 0 )
88 return;
89 uint32_t typeHdl( getHandle( inType ) );
90 uint32_t fileHdl( getHandle( inFile ) );
91 AllocationEvent theEvent;
92 theEvent.init( inSize, typeHdl, fileHdl, inLine, addr );
93 sendEvent( theEvent );
94 }
95
96 void onDeallocation( uint64_t addr )
97 {
98 if ( addr == 0 )
99 return;
100 DeallocationEvent theEvent;
101 theEvent.init( addr );
102 sendEvent( theEvent );
103 }
104
105 void flushProfileEvents()
106 {
107 TBaseType::flushEvents();
108 }
109
110 protected:
111
112 template<typename TDataType>
113 void sendEvent( TDataType inType )
114 {
115 MemoryEventHeader theHeader( getMemoryEventType<TDataType>() );
116 inType.setup( theHeader );
117 theHeader.streamify( TBaseType::mSerializer );
118 inType.streamify( TBaseType::mSerializer, theHeader );
119 if ( TBaseType::mDataArray.size() >= TBaseType::mBufferFullAmount )
120 flushProfileEvents();
121 }
122 };
