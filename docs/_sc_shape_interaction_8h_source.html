191PX_FORCE_INLINE void Sc::ShapeInteraction::sendLostTouchReport(bool shapeVolumeRemoved, const PxU32 ccdPass, PxsContactManagerOutputIterator& outputs)
192{
193 PX_ASSERT(hasTouch());
194 PX_ASSERT(isReportPair());
195
196 const PxU32 pairFlags = getPairFlags();
197 const PxU32 notifyTouchLost = pairFlags & PxU32(PxPairFlag::eNOTIFY_TOUCH_LOST);
198 const PxIntBool thresholdExceeded = readFlag(ShapeInteraction::FORCE_THRESHOLD_EXCEEDED_NOW);
199 const PxU32 notifyThresholdLost = thresholdExceeded ? (pairFlags & PxU32(PxPairFlag::eNOTIFY_THRESHOLD_FORCE_LOST)) : 0;
200 if(!notifyTouchLost && !notifyThresholdLost)
201 return;
202
203 PxU16 infoFlag = 0;
204 if (mActorPair->getTouchCount() == 1) // this code assumes that the actor pair touch count does get decremented afterwards
205 {
206 infoFlag |= PxContactPairFlag::eACTOR_PAIR_LOST_TOUCH;
207 }
208
209 //Lost touch is processed after solver, so we should use the previous transform to update the pose for objects if user request eCONTACT_EVENT_POSE
210 const bool useCurrentTransform = false;
211
212 const PxU32 triggeredFlags = notifyTouchLost | notifyThresholdLost;
213 PX_ASSERT(triggeredFlags);
214 processUserNotification(triggeredFlags, infoFlag, true, ccdPass, useCurrentTransform, outputs);
215
216 if(shapeVolumeRemoved)
217 {
218 ActorPairReport& apr = getActorPairReport();
219 ContactStreamManager& cs = apr.getContactStreamManager();
220 cs.raiseFlags(ContactStreamManagerFlag::eTEST_FOR_REMOVED_SHAPES);
221 }
222}

224PX_FORCE_INLINE void Sc::ShapeInteraction::setPairFlags(PxPairFlags flags)
225{
226 PX_ASSERT(PxU32(flags) < PxPairFlag::eNEXT_FREE); // to find out if a new PxPairFlag has been added after eLAST instead of in front
227
228 PxU32 newFlags = mFlags;
229 PxU32 fl = PxU32(flags) & PAIR_FLAGS_MASK;
230 newFlags &= (~PAIR_FLAGS_MASK); // clear old flags
231 newFlags |= fl;
232
233 mFlags = newFlags;
234}

242PX_INLINE void Sc::ShapeInteraction::swapAndClearForceThresholdExceeded()
243{
244 PxU32 flags = mFlags;
245
246 PX_COMPILE_TIME_ASSERT(FORCE_THRESHOLD_EXCEEDED_NOW == (FORCE_THRESHOLD_EXCEEDED_BEFORE >> 1));
247
248 PxU32 nowToBefore = (flags & FORCE_THRESHOLD_EXCEEDED_NOW) << 1;
249 flags &= ~(FORCE_THRESHOLD_EXCEEDED_NOW | FORCE_THRESHOLD_EXCEEDED_BEFORE);
250 flags |= nowToBefore;
251
252 mFlags = flags;
253}

255PX_FORCE_INLINE void Sc::ShapeInteraction::removeFromReportPairList()
256{
257 // this method should only get called if the pair is in the list for
258 // persistent or force based contact reports
259 PX_ASSERT(mReportPairIndex != INVALID_REPORT_PAIR_ID);
260 PX_ASSERT(readFlag(IS_IN_CONTACT_EVENT_LIST));
261
262 Scene& scene = getScene();
263
264 if (readFlag(IS_IN_FORCE_THRESHOLD_EVENT_LIST))
265 scene.getNPhaseCore()->removeFromForceThresholdContactEventPairs(this);
266 else
267 {
268 PX_ASSERT(readFlag(IS_IN_PERSISTENT_EVENT_LIST));
269 scene.getNPhaseCore()->removeFromPersistentContactEventPairs(this);
270 }

302PX_FORCE_INLINE bool Sc::ShapeInteraction::activeManagerAllowed() const
303{
304 ShapeSimBase& shape0 = getShape0();
305 ShapeSimBase& shape1 = getShape1();
306
307 ActorSim& bodySim0 = shape0.getActor();
308 ActorSim& bodySim1 = shape1.getActor();
309
310 // the first shape always belongs to a dynamic body or soft body
311 PX_ASSERT(bodySim0.isDynamicRigid() || bodySim0.isSoftBody() || bodySim0.isFEMCloth() || bodySim0.isParticleSystem() || bodySim0.isHairSystem());
312
313 const IG::IslandSim& islandSim = getScene().getSimpleIslandManager()->getSpeculativeIslandSim();
314
315 //check whether active in the speculative sim!
316
317 return (islandSim.getNode(bodySim0.getNodeIndex()).isActive() ||
318 (!bodySim1.isStaticRigid() && islandSim.getNode(bodySim1.getNodeIndex()).isActive()));
319}

340PX_FORCE_INLINE PxIntBool Sc::ShapeInteraction::hasKnownTouchState() const
341{
342 // For a pair where the bodies were added asleep, the touch state is not known until narrowphase runs on the pair for the first time.
343 // If such a pair looses AABB overlap before, the conservative approach is to wake the bodies up. This method provides an indicator that
344 // this is such a pair. Note: this might also wake up objects that do not touch but that's the price to pay (unless we want to run
345 // overlap tests on such pairs).
346 if (mManager)
347 return mManager->touchStatusKnown();
348 else
349 return readFlag(TOUCH_KNOWN);
350}
