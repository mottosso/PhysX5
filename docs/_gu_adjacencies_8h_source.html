178 PX_INLINE const AdjTriangle* GetAdjacentFace(const AdjTriangle& current_tri, SharedEdgeIndex edge_nb) const
179 {
180 // No checkings here, make sure mFaces has been created
181
182 // Catch the link
183 PxU32 Link = current_tri.GetLink(edge_nb);
184
185 // Returns NULL for boundary edges
186 if(IS_BOUNDARY(Link)) return NULL;
187
188 // Else transform into face index
189 PxU32 Id = MAKE_ADJ_TRI(Link);
190
191 // Possible counterpart edge is:
192 // PxU32 Edge = GET_EDGE_NB(Link);
193
194 // And returns adjacent triangle
195 return &mFaces[Id];
196 }
