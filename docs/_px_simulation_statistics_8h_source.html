203 PxU32 getRbPairStats(RbPairStatsType pairType, PxGeometryType::Enum g0, PxGeometryType::Enum g1) const
204 {
205 PX_ASSERT_WITH_MESSAGE( (pairType >= eDISCRETE_CONTACT_PAIRS) &&
206 (pairType <= eTRIGGER_PAIRS),
207 "Invalid pairType in PxSimulationStatistics::getRbPairStats");
208
209 if (g0 >= PxGeometryType::eGEOMETRY_COUNT || g1 >= PxGeometryType::eGEOMETRY_COUNT)
210 {
211 PX_ASSERT(false);
212 return 0;
213 }
214
215 PxU32 nbPairs = 0;
216 switch(pairType)
217 {
218 case eDISCRETE_CONTACT_PAIRS:
219 nbPairs = nbDiscreteContactPairs[g0][g1];
220 break;
221 case eCCD_PAIRS:
222 nbPairs = nbCCDPairs[g0][g1];
223 break;
224 case eMODIFIED_CONTACT_PAIRS:
225 nbPairs = nbModifiedContactPairs[g0][g1];
226 break;
227 case eTRIGGER_PAIRS:
228 nbPairs = nbTriggerPairs[g0][g1];
229 break;
230 }
231 return nbPairs;
232 }
