192 void setToRestState()
193 {
194 for(PxU32 i=0;i<4;i++)
195 {
196 mWheelSpeeds[i] = 0.0f;
197 mCorrectedWheelSpeeds[i] = 0.0f;
198 mWheelRotationAngles[i] = 0.0f;
199 mTireLowForwardSpeedTimers[i] = 0.0f;
200 mTireLowSideSpeedTimers[i] = 0.0f;
201 mJounces[i] = PX_MAX_F32;
202 mSteerAngles[i] = 0.0f;
203 mVehicleConstraints->mData.mStickyTireForwardData.mActiveFlags[i] = false;
204 mVehicleConstraints->mData.mStickyTireSideData.mActiveFlags[i] = false;
205 mVehicleConstraints->mData.mSuspLimitData.mActiveFlags[i] = false;
206 }
207 PxMemZero(mQueryOrCachedHitResults, sizeof(SuspLineSweep));

217 {
218 mWheelSpeeds[i] = 0.0f;
219 mCorrectedWheelSpeeds[i] = 0.0f;
220 mJounces[i] = PX_MAX_F32; //Ensure that the jounce speed is zero when the car wakes up again.

225 void setTireContacts(const PxU32* cachedHitCounts, const PxPlane* cachedHitPlanes, const PxF32* cachedFrictionMultipliers, const PxTireContactIntersectionMethod::Enum* cachedQueryTypes)
226 {
227 mHasCachedRaycastHitPlane = true;
228 mRaycastResults = NULL;
229 mSweepResults = NULL;
230
231 PxVehicleWheels4DynData::CachedSuspLineSceneQuerytHitResult* cachedRaycastHitResults =
232 reinterpret_cast<PxVehicleWheels4DynData::CachedSuspLineSceneQuerytHitResult*>(mQueryOrCachedHitResults);
233
234 for (PxU32 i = 0; i < 4; i++)
235 {
236 cachedRaycastHitResults->mCounts[i] = PxU16(cachedHitCounts[i]);
237 cachedRaycastHitResults->mPlanes[i] = cachedHitPlanes[i];
238 cachedRaycastHitResults->mDistances[i] = 1.0f;
239 cachedRaycastHitResults->mFrictionMultipliers[i] = cachedFrictionMultipliers[i];
240 cachedRaycastHitResults->mQueryTypes[i] = ((cachedQueryTypes[i] == PxTireContactIntersectionMethod::eRAY) ? 0 : 1);
241 }
242 }
