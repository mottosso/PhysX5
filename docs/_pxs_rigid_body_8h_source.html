46 enum PxsRigidBodyFlag

153void PxsRigidBody::constrainLinearVelocity()
154{
155 const PxU32 lockFlags = mCore->lockFlags;
156 if(lockFlags)
157 {
158 if(lockFlags & PxRigidDynamicLockFlag::eLOCK_LINEAR_X)
159 mCore->linearVelocity.x = 0.0f;
160 if(lockFlags & PxRigidDynamicLockFlag::eLOCK_LINEAR_Y)
161 mCore->linearVelocity.y = 0.0f;
162 if(lockFlags & PxRigidDynamicLockFlag::eLOCK_LINEAR_Z)
163 mCore->linearVelocity.z = 0.0f;
164 }
165}

167void PxsRigidBody::constrainAngularVelocity()
168{
169 const PxU32 lockFlags = mCore->lockFlags;
170 if(lockFlags)
171 {
172 if(lockFlags & PxRigidDynamicLockFlag::eLOCK_ANGULAR_X)
173 mCore->angularVelocity.x = 0.0f;
174 if(lockFlags & PxRigidDynamicLockFlag::eLOCK_ANGULAR_Y)
175 mCore->angularVelocity.y = 0.0f;
176 if(lockFlags & PxRigidDynamicLockFlag::eLOCK_ANGULAR_Z)
177 mCore->angularVelocity.z = 0.0f;
178 }
179}
