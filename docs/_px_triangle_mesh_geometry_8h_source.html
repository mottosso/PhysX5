141PX_INLINE bool PxTriangleMeshGeometry::isValid() const
142{
143 if(mType != PxGeometryType::eTRIANGLEMESH)
144 return false;
145 if(!scale.scale.isFinite() || !scale.rotation.isUnit())
146 return false;
147 if(!scale.isValidForTriangleMesh())
148 return false;
149 if(!triangleMesh)
150 return false;
151
152 return true;
153}
