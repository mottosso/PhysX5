314 PX_CUDA_CALLABLE PX_FORCE_INLINE PxContactStreamIterator(const PxU8* contactPatches, const PxU8* contactPoints, const PxU32* contactFaceIndices, PxU32 nbPatches, PxU32 nbContacts)
315 : zero(0.f)
316 {
317 bool modify = false;
318 bool compressedModify = false;
319 bool response = false;
320 bool indices = false;
321
322 PxU32 pointSize = 0;
323 PxU32 patchHeaderSize = sizeof(PxContactPatch);
324
325 const PxContactPatch* patches = reinterpret_cast<const PxContactPatch*>(contactPatches);
326
327 if(patches)
328 {
329 modify = (patches->internalFlags & PxContactPatch::eMODIFIABLE) != 0;
330 compressedModify = (patches->internalFlags & PxContactPatch::eCOMPRESSED_MODIFIED_CONTACT) != 0;
331 indices = (patches->internalFlags & PxContactPatch::eHAS_FACE_INDICES) != 0;
332
333 patch = patches;
334
335 contact = reinterpret_cast<const PxContact*>(contactPoints);
336
337 faceIndice = contactFaceIndices;
338
339 pointSize = compressedModify ? sizeof(PxExtendedContact) : modify ? sizeof(PxModifiableContact) : sizeof(PxContact);
340
341 response = (patch->internalFlags & PxContactPatch::eFORCE_NO_RESPONSE) == 0;
342 }
343
344
345 mStreamFormat = compressedModify ? eCOMPRESSED_MODIFIABLE_STREAM : modify ? eMODIFIABLE_STREAM : eSIMPLE_STREAM;
346 hasFaceIndices = PxU32(indices);
347 forceNoResponse = PxU32(!response);
348
349 contactPatchHeaderSize = patchHeaderSize;
350 contactPointSize = pointSize;
351 nextPatchIndex = 0;
352 nextContactIndex = 0;
353 totalContacts = nbContacts;
354 totalPatches = nbPatches;
355
356 pointStepped = false;
357 }

389 PX_CUDA_CALLABLE PX_INLINE void nextPatch()
390 {
391 PX_ASSERT(nextPatchIndex < totalPatches);
392 if(nextPatchIndex)
393 {
394 if(nextContactIndex < patch->nbContacts)
395 {
396 PxU32 nbToStep = patch->nbContacts - this->nextContactIndex;
397 contact = reinterpret_cast<const PxContact*>(reinterpret_cast<const PxU8*>(contact) + contactPointSize * nbToStep);
398 }
399 patch = reinterpret_cast<const PxContactPatch*>(reinterpret_cast<const PxU8*>(patch) + contactPatchHeaderSize);
400 }
401 nextPatchIndex++;
402 nextContactIndex = 0;
403 }

423 faceIndice++;
424 }

595 bool advanceToIndex(const PxU32 initialIndex)
596 {
597 PX_ASSERT(this->nextPatchIndex == 0 && this->nextContactIndex == 0);
598
599 PxU32 numToAdvance = initialIndex;
600
601 if(numToAdvance == 0)
602 {
603 PX_ASSERT(hasNextPatch());
604 nextPatch();
605 return true;
606 }
607
608 while(numToAdvance)
609 {
610 while(hasNextPatch())
611 {
612 nextPatch();
613 PxU32 patchSize = patch->nbContacts;
614 if(numToAdvance <= patchSize)
615 {
616 contact = reinterpret_cast<const PxContact*>(reinterpret_cast<const PxU8*>(contact) + contactPointSize * numToAdvance);
617 nextContactIndex += numToAdvance;
618 return true;
619 }
620 else
621 {
622 numToAdvance -= patchSize;
623 }
624 }
625 }
626 return false;
627 }
