606PX_INLINE bool PxControllerDesc::isValid() const
607{
608 if( mType!=PxControllerShapeType::eBOX
609 && mType!=PxControllerShapeType::eCAPSULE)
610 return false;
611 if(scaleCoeff<0.0f)
612 return false;
613 if(volumeGrowth<1.0f)
614 return false;
615 if(density<0.0f)
616 return false;
617 if(slopeLimit<0.0f)
618 return false;
619 if(stepOffset<0.0f)
620 return false;
621 if(contactOffset<=0.0f)
622 return false;
623 if(!material)
624 return false;
625 if(!toVec3(position).isFinite())
626 return false; //the float version needs to be finite otherwise actor creation will fail.
627
628 return true;
629}
