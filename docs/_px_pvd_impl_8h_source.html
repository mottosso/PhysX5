56struct MetaDataProvider : public PvdOMMetaDataProvider, public PxUserAllocated
57{
58 typedef PxMutex::ScopedLock TScopedLockType;
59 typedef PxHashMap<const void*, int32_t> TInstTypeMap;
60 PvdObjectModelMetaData& mMetaData;
61 PxMutex mMutex;
62 uint32_t mRefCount;
63 TInstTypeMap mTypeMap;
64
65 MetaDataProvider()
66 : mMetaData(PvdObjectModelMetaData::create()), mRefCount(0), mTypeMap("MetaDataProvider::mTypeMap")
67 {
68 mMetaData.addRef();
69 }
70 virtual ~MetaDataProvider()
71 {
72 mMetaData.release();
73 }
74
75 virtual void addRef()
76 {
77 TScopedLockType locker(mMutex);
78 ++mRefCount;
79 }
80 virtual void release()
81 {
82 {
83 TScopedLockType locker(mMutex);
84 if(mRefCount)
85 --mRefCount;
86 }
87 if(!mRefCount)
88 PVD_DELETE(this);
89 }
90 virtual PvdObjectModelMetaData& lock()
91 {
92 mMutex.lock();
93 return mMetaData;
94 }
95 virtual void unlock()
96 {
97 mMutex.unlock();
98 }
99
100 virtual bool createInstance(const NamespacedName& clsName, const void* instance)
101 {
102 TScopedLockType locker(mMutex);
103 Option<ClassDescription> cls(mMetaData.findClass(clsName));
104 if(cls.hasValue() == false)
105 return false;
106 int32_t instType = cls->mClassId;
107 mTypeMap.insert(instance, instType);
108 return true;
109 }
110 virtual bool isInstanceValid(const void* instance)
111 {
112 TScopedLockType locker(mMutex);
113 ClassDescription classDesc;
114 bool retval = mTypeMap.find(instance) != NULL;
115#if PX_DEBUG
116 if(retval)
117 classDesc = mMetaData.getClass(mTypeMap.find(instance)->second);
118#endif
119 return retval;
120 }
121 virtual void destroyInstance(const void* instance)
122 {
123 {
124 TScopedLockType locker(mMutex);
125 mTypeMap.erase(instance);
126 }
127 }
128 virtual int32_t getInstanceClassType(const void* instance)
129 {
130 TScopedLockType locker(mMutex);
131 const TInstTypeMap::Entry* entry = mTypeMap.find(instance);
132 if(entry)
133 return entry->second;
134 return -1;
135 }
136
137 private:
138 MetaDataProvider& operator=(const MetaDataProvider&);
139 MetaDataProvider(const MetaDataProvider&);
140};
