97#define SERIAL_OBJECT_INDEX_TYPE_BIT (1u<<31)
98 struct SerialObjectIndex
99 {
100 //= ATTENTION! =====================================================================================
101 // Changing the data layout of this class breaks the binary serialization format. See comments for
102 // PX_BINARY_SERIAL_VERSION. If a modification is required, please adjust the getBinaryMetaData
103 // function. If the modification is made on a custom branch, please change PX_BINARY_SERIAL_VERSION
104 // accordingly.
105 //==================================================================================================
106
107 PX_FORCE_INLINE SerialObjectIndex(PxU32 index, bool external) { setIndex(index, external); }
108 PX_FORCE_INLINE SerialObjectIndex(const SerialObjectIndex& objIndex) : mObjIndex(objIndex.mObjIndex) {}
109 PX_FORCE_INLINE SerialObjectIndex() : mObjIndex(PX_INVALID_U32) {}
110
111 PX_FORCE_INLINE void setIndex(PxU32 index, bool external)
112 {
113 PX_ASSERT((index & SERIAL_OBJECT_INDEX_TYPE_BIT) == 0);
114 mObjIndex = index | (external ? SERIAL_OBJECT_INDEX_TYPE_BIT : 0);
115 }
116
117 PX_FORCE_INLINE PxU32 getIndex(bool& isExternal)
118 {
119 PX_ASSERT(mObjIndex != PX_INVALID_U32);
120 isExternal = (mObjIndex & SERIAL_OBJECT_INDEX_TYPE_BIT) > 0;
121 return mObjIndex & ~SERIAL_OBJECT_INDEX_TYPE_BIT;
122 }
123
124 PX_FORCE_INLINE bool operator < (const SerialObjectIndex& so) const
125 {
126 return mObjIndex < so.mObjIndex;
127 }
128
129 private:
130 PxU32 mObjIndex;
131 };

264 virtual void writeData(const void* buffer, PxU32 size) { mMemStream.write(buffer, size); }
265 virtual PxU32 getTotalStoredSize() { return mMemStream.getSize(); }
266 virtual void alignData(PxU32 alignment = PX_SERIAL_ALIGN)
267 {
268 if(!alignment)
269 return;
270
271 PxI32 bytesToPad = PxI32(getPadding(mMemStream.getSize(), alignment));
272 static const PxI32 BUFSIZE = 64;
273 char buf[BUFSIZE];
274 PxMemSet(buf, 0, bytesToPad < BUFSIZE ? PxU32(bytesToPad) : PxU32(BUFSIZE));
275 while(bytesToPad > 0)
276 {
277 mMemStream.write(buf, bytesToPad < BUFSIZE ? PxU32(bytesToPad) : PxU32(BUFSIZE));
278 bytesToPad -= BUFSIZE;
279 }
280 PX_ASSERT(!getPadding(getTotalStoredSize(), alignment));
281 }
