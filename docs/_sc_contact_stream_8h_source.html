221PX_FORCE_INLINE void Sc::ContactStreamManager::convertDeletedShapesInContactStream(ContactShapePair* shapePairs, PxU32 pairCount, const ObjectIDTracker& tracker)
222{
223 for(PxU32 i=0; i < pairCount; i++)
224 {
225 ContactShapePair& csp = shapePairs[i];
226
227 PxU32 shape0ID = csp.shapeID[0];
228 PxU32 shape1ID = csp.shapeID[1];
229
230 PxU16 flags = csp.flags;
231 PX_COMPILE_TIME_ASSERT(sizeof(flags) == sizeof((reinterpret_cast<ContactShapePair*>(0))->flags));
232
233 if (tracker.isDeletedID(shape0ID))
234 flags |= PxContactPairFlag::eREMOVED_SHAPE_0;
235 if (tracker.isDeletedID(shape1ID))
236 flags |= PxContactPairFlag::eREMOVED_SHAPE_1;
237
238 csp.flags = flags;
239 }
240}

243PX_FORCE_INLINE PxU32 Sc::ContactStreamManager::computeExtraDataBlockCount(PxU32 extraDataSize_)
244{
245 PxU32 nbBlocks;
246 if (extraDataSize_ & ((1 << sExtraDataBlockSizePow2) - 1)) // not a multiple of block size -> need one block more
247 nbBlocks = (extraDataSize_ >> sExtraDataBlockSizePow2) + 1;
248 else
249 nbBlocks = (extraDataSize_ >> sExtraDataBlockSizePow2);
250
251 return nbBlocks;
252}

261PX_FORCE_INLINE PxU16 Sc::ContactStreamManager::computeContactReportExtraDataSize(PxU32 extraDataFlags, bool addHeader)
262{
263 PX_ASSERT(extraDataFlags);
264
265 PxU16 extraDataSize_ = sizeof(PxContactPairIndex);
266 if (extraDataFlags & PxPairFlag::ePRE_SOLVER_VELOCITY)
267 extraDataSize_ += sizeof(PxContactPairVelocity);
268 if (extraDataFlags & PxPairFlag::ePOST_SOLVER_VELOCITY)
269 extraDataSize_ += sizeof(PxContactPairVelocity);
270 if (extraDataFlags & PxPairFlag::eCONTACT_EVENT_POSE)
271 extraDataSize_ += sizeof(PxContactPairPose);
272 if (addHeader)
273 extraDataSize_ += sizeof(ContactStreamHeader);
274 return extraDataSize_;
275}

278PX_FORCE_INLINE void Sc::ContactStreamManager::fillInContactReportExtraData(PxContactPairVelocity* cpVel, PxU32 index, const ActorSim& rs, bool isCCDPass)
279{
280 if (rs.getActorType() != PxActorType::eRIGID_STATIC)
281 {
282 const BodySim& bs = static_cast<const BodySim&>(rs);
283 if ((!isCCDPass) || (cpVel->type == PxContactPairExtraDataType::ePOST_SOLVER_VELOCITY))
284 {
285 const BodyCore& bc = bs.getBodyCore();
286 cpVel->linearVelocity[index] = bc.getLinearVelocity();
287 cpVel->angularVelocity[index] = bc.getAngularVelocity();
288 }
289 else
290 {
291 PX_ASSERT(cpVel->type == PxContactPairExtraDataType::ePRE_SOLVER_VELOCITY);
292 const Cm::SpatialVector& vel = bs.getLowLevelBody().getPreSolverVelocities();
293 cpVel->linearVelocity[index] = vel.linear;
294 cpVel->angularVelocity[index] = vel.angular;
295 }
296 }
297 else
298 {
299 cpVel->linearVelocity[index] = PxVec3(0.0f);
300 cpVel->angularVelocity[index] = PxVec3(0.0f);
301 }
302}

305PX_FORCE_INLINE void Sc::ContactStreamManager::fillInContactReportExtraData(PxContactPairPose* cpPose, PxU32 index, const ActorSim& rs, bool isCCDPass, const bool useCurrentTransform)
306{
307 if(rs.getActorType() != PxActorType::eRIGID_STATIC)
308 {
309 const BodySim& bs = static_cast<const BodySim&>(rs);
310 const BodyCore& bc = bs.getBodyCore();
311 const PxTransform& src = (!isCCDPass && useCurrentTransform) ? bc.getBody2World() : bs.getLowLevelBody().getLastCCDTransform();
312 cpPose->globalPose[index] = src * bc.getBody2Actor().getInverse();
313 }
314 else
315 {
316 const StaticSim& ss = static_cast<const StaticSim&>(rs);
317 const StaticCore& sc = ss.getStaticCore();
318 cpPose->globalPose[index] = sc.getActor2World();
319 }
320}

323PX_FORCE_INLINE void Sc::ContactStreamManager::fillInContactReportExtraData(PxU8* stream, PxU32 extraDataFlags, const ActorSim& rs0, const ActorSim& rs1, PxU32 ccdPass, const bool useCurrentTransform,
324 PxU32 pairIndex, PxU32 sizeOffset)
325{
326 ContactStreamHeader* strHeader = reinterpret_cast<ContactStreamHeader*>(stream);
327 strHeader->contactPass = PxTo16(ccdPass);
328
329 stream += sizeOffset;
330 PxU8* edStream = stream;
331 bool isCCDPass = (ccdPass != 0);
332
333 {
334 PxContactPairIndex* cpIndex = reinterpret_cast<PxContactPairIndex*>(edStream);
335 cpIndex->type = PxContactPairExtraDataType::eCONTACT_PAIR_INDEX;
336 cpIndex->index = PxTo16(pairIndex);
337 edStream += sizeof(PxContactPairIndex);
338
339 PX_ASSERT(edStream <= reinterpret_cast<PxU8*>(getShapePairs(stream)));
340 }
341
342 // Important: make sure this one is the first after the PxContactPairIndex item for discrete contacts as it needs to get filled in before the reports get sent
343 // (post solver velocity is not available when it gets created)
344 if (extraDataFlags & PxPairFlag::ePOST_SOLVER_VELOCITY)
345 {
346 PxContactPairVelocity* cpVel = reinterpret_cast<PxContactPairVelocity*>(edStream);
347 cpVel->type = PxContactPairExtraDataType::ePOST_SOLVER_VELOCITY;
348 edStream += sizeof(PxContactPairVelocity);
349
350 if (!isCCDPass)
351 raiseFlags(ContactStreamManagerFlag::eNEEDS_POST_SOLVER_VELOCITY); // don't know the post solver velocity yet
352 else
353 {
354 ContactStreamManager::fillInContactReportExtraData(cpVel, 0, rs0, true);
355 ContactStreamManager::fillInContactReportExtraData(cpVel, 1, rs1, true);
356 }
357
358 PX_ASSERT(edStream <= reinterpret_cast<PxU8*>(getShapePairs(stream)));
359 }
360 if (extraDataFlags & PxPairFlag::ePRE_SOLVER_VELOCITY)
361 {
362 PxContactPairVelocity* cpVel = reinterpret_cast<PxContactPairVelocity*>(edStream);
363 cpVel->type = PxContactPairExtraDataType::ePRE_SOLVER_VELOCITY;
364 ContactStreamManager::fillInContactReportExtraData(cpVel, 0, rs0, isCCDPass);
365 ContactStreamManager::fillInContactReportExtraData(cpVel, 1, rs1, isCCDPass);
366 edStream += sizeof(PxContactPairVelocity);
367
368 PX_ASSERT(edStream <= reinterpret_cast<PxU8*>(getShapePairs(stream)));
369 }
370 if (extraDataFlags & PxPairFlag::eCONTACT_EVENT_POSE)
371 {
372 PxContactPairPose* cpPose = reinterpret_cast<PxContactPairPose*>(edStream);
373 cpPose->type = PxContactPairExtraDataType::eCONTACT_EVENT_POSE;
374 ContactStreamManager::fillInContactReportExtraData(cpPose, 0, rs0, isCCDPass, useCurrentTransform);
375 ContactStreamManager::fillInContactReportExtraData(cpPose, 1, rs1, isCCDPass, useCurrentTransform);
376 edStream += sizeof(PxContactPairPose);
377
378 PX_ASSERT(edStream <= reinterpret_cast<PxU8*>(getShapePairs(stream)));
379 }
380
381 extraDataSize = PxTo16(sizeOffset + PxU32(edStream - stream));
382}

385PX_FORCE_INLINE void Sc::ContactStreamManager::setContactReportPostSolverVelocity(PxU8* stream, const ActorSim& rs0, const ActorSim& rs1)
386{
387 PX_ASSERT(extraDataSize > (sizeof(ContactStreamHeader) + sizeof(PxContactPairIndex)));
388 PxContactPairVelocity* cpVel = reinterpret_cast<PxContactPairVelocity*>(stream + sizeof(ContactStreamHeader) + sizeof(PxContactPairIndex));
389 PX_ASSERT(cpVel->type == PxContactPairExtraDataType::ePOST_SOLVER_VELOCITY);
390
391 fillInContactReportExtraData(cpVel, 0, rs0, false);
392 fillInContactReportExtraData(cpVel, 1, rs1, false);
393
394 clearFlags(ContactStreamManagerFlag::eNEEDS_POST_SOLVER_VELOCITY);
395}
