63struct PxcDataStreamPool
64{
65 PxU8* mDataStream;
66 PxI32 mSharedDataIndex;
67 PxU32 mDataStreamSize;
68 PxU32 mSharedDataIndexGPU;
69
70 bool isOverflown() const
71 {
72 //FD: my expectaton is that reading those variables is atomic, shared indices are non-decreasing,
73 //so we can only get a false overflow alert because of concurrency issues, which is not a big deal as it means
74 //it did overflow a bit later
75 return (mSharedDataIndex + mSharedDataIndexGPU) > mDataStreamSize;
76 }
77};
