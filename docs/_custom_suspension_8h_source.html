66class CustomSuspensionComponent : public PxVehicleComponent
67{
68public:
69
70 CustomSuspensionComponent() : PxVehicleComponent() {}
71
72 virtual void getDataForCustomSuspensionComponent(
73 const PxVehicleAxleDescription*& axleDescription,
74 const PxVehicleRigidBodyParams*& rigidBodyParams,
75 const PxVehicleSuspensionStateCalculationParams*& suspensionStateCalculationParams,
76 const PxReal*& steerResponseStates,
77 const PxVehicleRigidBodyState*& rigidBodyState,
78 const PxVehicleWheelParams*& wheelParams,
79 const PxVehicleSuspensionParams*& suspensionParams,
80 const CustomSuspensionParams*& customSuspensionParams,
81 const PxVehicleSuspensionComplianceParams*& suspensionComplianceParams,
82 const PxVehicleSuspensionForceParams*& suspensionForceParams,
83 const PxVehicleRoadGeometryState*& wheelRoadGeomStates,
84 PxVehicleSuspensionState*& suspensionStates,
85 CustomSuspensionState*& customSuspensionStates,
86 PxVehicleSuspensionComplianceState*& suspensionComplianceStates,
87 PxVehicleSuspensionForce*& suspensionForces) = 0;
88
89 virtual bool update(const PxReal dt, const PxVehicleSimulationContext& context)
90 {
91 const PxVehicleAxleDescription* axleDescription;
92 const PxVehicleRigidBodyParams* rigidBodyParams;
93 const PxVehicleSuspensionStateCalculationParams* suspensionStateCalculationParams;
94 const PxReal* steerResponseStates;
95 const PxVehicleRigidBodyState* rigidBodyState;
96 const PxVehicleWheelParams* wheelParams;
97 const PxVehicleSuspensionParams* suspensionParams;
98 const CustomSuspensionParams* customSuspensionParams;
99 const PxVehicleSuspensionComplianceParams* suspensionComplianceParams;
100 const PxVehicleSuspensionForceParams* suspensionForceParams;
101 const PxVehicleRoadGeometryState* wheelRoadGeomStates;
102 PxVehicleSuspensionState* suspensionStates;
103 CustomSuspensionState* customSuspensionStates;
104 PxVehicleSuspensionComplianceState* suspensionComplianceStates;
105 PxVehicleSuspensionForce* suspensionForces;
106
107 getDataForCustomSuspensionComponent(axleDescription, rigidBodyParams, suspensionStateCalculationParams,
108 steerResponseStates, rigidBodyState, wheelParams, suspensionParams, customSuspensionParams,
109 suspensionComplianceParams, suspensionForceParams, wheelRoadGeomStates,
110 suspensionStates, customSuspensionStates, suspensionComplianceStates, suspensionForces);
111
112 for (PxU32 i = 0; i < axleDescription->nbWheels; i++)
113 {
114 const PxU32 wheelId = axleDescription->wheelIdsInAxleOrder[i];
115
116 //Update the suspension state (jounce, jounce speed)
117 PxVehicleSuspensionStateUpdate(
118 wheelParams[wheelId], suspensionParams[wheelId], *suspensionStateCalculationParams,
119 suspensionForceParams[wheelId].stiffness, suspensionForceParams[wheelId].damping,
120 steerResponseStates[wheelId], wheelRoadGeomStates[wheelId],
121 *rigidBodyState,
122 dt, context.frame, context.gravity,
123 suspensionStates[wheelId]);
124
125 //Update the compliance from the suspension state.
126 PxVehicleSuspensionComplianceUpdate(
127 suspensionParams[wheelId], suspensionComplianceParams[wheelId],
128 suspensionStates[wheelId],
129 suspensionComplianceStates[wheelId]);
130
131 //Compute the suspension force from the suspension and compliance states.
132 PxVehicleSuspensionForceUpdate(
133 suspensionParams[wheelId], suspensionForceParams[wheelId],
134 wheelRoadGeomStates[wheelId], suspensionStates[wheelId],
135 suspensionComplianceStates[wheelId], *rigidBodyState,
136 context.gravity, rigidBodyParams->mass,
137 suspensionForces[wheelId]);
138
139 addCustomSuspensionForce(dt,
140 suspensionParams[wheelId],
141 customSuspensionParams[wheelId],
142 wheelRoadGeomStates[wheelId], suspensionComplianceStates[wheelId], *rigidBodyState,
143 suspensionForces[wheelId], customSuspensionStates[wheelId]);
144 }
145
146 return true;
147 }
148};
