65 const char* getKey(uint32_t index) const
66 {
67 PX_ASSERT((index * 2) < uint32_t(argc));
68 return argv[index * 2];
69 }

71 const char* getValue(uint32_t index) const
72 {
73 PX_ASSERT((index * 2 + 1) < uint32_t(argc));
74 return argv[index * 2 + 1];
75 }
76
77 const char* get(const char* attr) const
78 {
79 int32_t count = argc / 2;
80 for(int32_t i = 0; i < count; ++i)
81 {
82 const char* key = argv[i * 2], *value = argv[i * 2 + 1];
83 if(strcmp(key, attr) == 0)
84 return value;
85 }
86
87 return NULL;
88 }
