755 PX_CUDA_CALLABLE bool operator<(const PxParticleRigidFilterPair& other) const
756 {
757 if(mID0 < other.mID0)
758 return true;
759 if(mID0 == other.mID0 && mID1 < other.mID1)
760 return true;
761 return false;
762 }
763
764 PX_CUDA_CALLABLE bool operator>(const PxParticleRigidFilterPair& other) const
765 {
766 if(mID0 > other.mID0)
767 return true;
768 if(mID0 == other.mID0 && mID1 > other.mID1)
769 return true;
770 return false;
771 }
772
773 PX_CUDA_CALLABLE bool operator==(const PxParticleRigidFilterPair& other) const
774 {
775 return (mID0 == other.mID0 && mID1 == other.mID1);
776 }
