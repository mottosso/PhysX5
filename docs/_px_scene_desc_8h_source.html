967PX_INLINE bool PxSceneDesc::isValid() const
968{
969 if(!PxSceneQueryDesc::isValid())
970 return false;
971
972 if(!filterShader)
973 return false;
974
975 if( ((filterShaderDataSize == 0) && (filterShaderData != NULL)) ||
976 ((filterShaderDataSize > 0) && (filterShaderData == NULL)) )
977 return false;
978
979 if(!limits.isValid())
980 return false;
981
982 if(bounceThresholdVelocity <= 0.0f)
983 return false;
984 if(frictionOffsetThreshold < 0.0f)
985 return false;
986 if(frictionCorrelationDistance <= 0)
987 return false;
988
989 if(maxBiasCoefficient < 0.0f)
990 return false;
991
992 if(!ccdMaxPasses)
993 return false;
994 if(ccdThreshold <= 0.0f)
995 return false;
996 if(ccdMaxSeparation < 0.0f)
997 return false;
998
999 if(!cpuDispatcher)
1000 return false;
1001
1002 if(!contactReportStreamBufferSize)
1003 return false;
1004
1005 if(maxNbContactDataBlocks < nbContactDataBlocks)
1006 return false;
1007
1008 if(wakeCounterResetValue <= 0.0f)
1009 return false;
1010
1011 if(!sanityBounds.isValid())
1012 return false;
1013
1014#if PX_SUPPORT_GPU_PHYSX
1015 if(!PxIsPowerOfTwo(gpuMaxNumPartitions))
1016 return false;
1017 if(gpuMaxNumPartitions > 32)
1018 return false;
1019 if (gpuMaxNumPartitions == 0)
1020 return false;
1021 if(!gpuDynamicsConfig.isValid())
1022 return false;
1023
1024 if (flags & PxSceneFlag::eSUPPRESS_READBACK)
1025 {
1026 if(!(flags & PxSceneFlag::eENABLE_GPU_DYNAMICS && broadPhaseType == PxBroadPhaseType::eGPU))
1027 return false;
1028 }
1029#endif
1030
1031 if(contactPairSlabSize == 0)
1032 return false;
1033
1034 return true;
1035}
