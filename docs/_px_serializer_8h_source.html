192 virtual void exportData(PxBase& obj, PxSerializationContext& s) const
193 {
194 PxAllocatorCallback& allocator = *PxGetAllocatorCallback();
195 T* copy = reinterpret_cast<T*>(allocator.allocate(sizeof(T), "TmpAllocExportData", __FILE__, __LINE__));
196 PxMemCopy(copy, &obj, sizeof(T));
197 copy->preExportDataReset();
198 s.writeData(copy, sizeof(T));
199 allocator.deallocate(copy);
200 }
201
202 virtual void registerReferences(PxBase& obj, PxSerializationContext& s) const
203 {
204 T& t = static_cast<T&>(obj);
205
206 s.registerReference(obj, PX_SERIAL_REF_KIND_PXBASE, size_t(&obj));
207
208 struct RequiresCallback : public PxProcessPxBaseCallback
209 {
210 RequiresCallback(PxSerializationContext& c) : context(c) {}
211 RequiresCallback& operator=(RequiresCallback&) { PX_ASSERT(0); return *this; }
212 void process(physx::PxBase& base)
213 {
214 context.registerReference(base, PX_SERIAL_REF_KIND_PXBASE, size_t(&base));
215 }
216 PxSerializationContext& context;
217 };
218
219 RequiresCallback callback(s);
220 t.requiresObjects(callback);
221 }
