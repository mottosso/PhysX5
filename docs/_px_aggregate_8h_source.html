58 // PxAggregateFilterHint is used for more efficient filtering of aggregates outside of the broadphase.
59 // It is a combination of a PxAggregateType and a self-collision bit.
60 typedef PxU32 PxAggregateFilterHint;
61
62 PX_CUDA_CALLABLE PX_FORCE_INLINE PxAggregateFilterHint PxGetAggregateFilterHint(PxAggregateType::Enum type, bool enableSelfCollision)
63 {
64 const PxU32 selfCollisionBit = enableSelfCollision ? 1 : 0;
65 return PxAggregateFilterHint((PxU32(type)<<1)|selfCollisionBit);
66 }
67
68 PX_CUDA_CALLABLE PX_FORCE_INLINE PxU32 PxGetAggregateSelfCollisionBit(PxAggregateFilterHint hint)
69 {
70 return hint & 1;
71 }
72
73 PX_CUDA_CALLABLE PX_FORCE_INLINE PxAggregateType::Enum PxGetAggregateType(PxAggregateFilterHint hint)
74 {
75 return PxAggregateType::Enum(hint>>1);
76 }
