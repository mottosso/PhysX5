145 virtual bool canRun(PvdInstanceDataStream &inStream_ )
146 {
147 PX_ASSERT(inStream_.isInstanceValid(&mJoint));
148 //When run this command, the constraint maybe buffer removed
149 return ((actor0 == NULL) || inStream_.isInstanceValid(actor0))
150 && ((actor1 == NULL) || inStream_.isInstanceValid(actor1));
151 }
152 virtual void run( PvdInstanceDataStream &inStream_ )
153 {
154 //When run this command, the constraint maybe buffer removed
155 if(!inStream_.isInstanceValid(&mJoint))
156 return;
157
158 PxRigidActor* actor0_, *actor1_;
159 mConstraint.getActors( actor0_, actor1_ );
160
161 if ( actor0_ && (inStream_.isInstanceValid(actor0_)) )
162 inStream_.pushBackObjectRef( actor0_, "Joints", &mJoint );
163 if ( actor1_ && (inStream_.isInstanceValid(actor1_)) )
164 inStream_.pushBackObjectRef( actor1_, "Joints", &mJoint );
165 const void* parent = actor0_ ? actor0_ : actor1_;
166 inStream_.setPropertyValue( &mJoint, "Parent", parent );
167 }
168 };
169
170 ConstraintUpdateCmd* cmd = PX_PLACEMENT_NEW(inStream.allocateMemForCmd(sizeof(ConstraintUpdateCmd)),
171 ConstraintUpdateCmd)(c, inSource);
172
173 if(cmd->canRun( inStream ))
174 cmd->run( inStream );
175 else
176 inStream.pushPvdCommand( *cmd );
177 }
