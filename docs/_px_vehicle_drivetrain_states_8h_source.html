194 void addConstraintGroup(const PxU32 nbWheelsInGroupToAdd, const PxU32* const wheelIdsInGroupToAdd, const PxF32* constraintMultipliers)
195 {
196 PX_ASSERT((nbWheelsInGroups + nbWheelsInGroupToAdd) < PxVehicleLimits::eMAX_NB_WHEELS);
197 PX_ASSERT(nbGroups < PxVehicleLimits::eMAX_NB_WHEELS);
198 nbWheelsPerGroup[nbGroups] = nbWheelsInGroupToAdd;
199 groupToWheelIds[nbGroups] = nbWheelsInGroups;
200 for (PxU32 i = 0; i < nbWheelsInGroupToAdd; i++)
201 {
202 wheelIdsInGroupOrder[nbWheelsInGroups + i] = wheelIdsInGroupToAdd[i];
203 wheelMultipliersInGroupOrder[nbWheelsInGroups + i] = constraintMultipliers[i];
204 }
205 nbWheelsInGroups += nbWheelsInGroupToAdd;
206 nbGroups++;
207 }
208
214 PX_FORCE_INLINE PxU32 getNbConstraintGroups() const
215 {
216 return nbGroups;
217 }
218
225 PX_FORCE_INLINE PxU32 getNbWheelsInConstraintGroup(const PxU32 i) const
226 {
227 return nbWheelsPerGroup[i];
228 }
229
237 PX_FORCE_INLINE PxU32 getWheelInConstraintGroup(const PxU32 j, const PxU32 i) const
238 {
239 return wheelIdsInGroupOrder[groupToWheelIds[i] + j];
240 }
241
248 PX_FORCE_INLINE PxReal getMultiplierInConstraintGroup(const PxU32 j, const PxU32 i) const
249 {
250 return wheelMultipliersInGroupOrder[groupToWheelIds[i] + j];
251 }
252
253 PxU32 nbGroups;
254 PxU32 nbWheelsPerGroup[PxVehicleLimits::eMAX_NB_AXLES];
255 PxU32 groupToWheelIds[PxVehicleLimits::eMAX_NB_AXLES];
256
257 PxU32 wheelIdsInGroupOrder[PxVehicleLimits::eMAX_NB_WHEELS];
258 PxF32 wheelMultipliersInGroupOrder[PxVehicleLimits::eMAX_NB_WHEELS];
259 PxU32 nbWheelsInGroups;
260};
