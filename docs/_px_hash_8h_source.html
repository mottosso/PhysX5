52// Hash functions
53
54// Thomas Wang's 32 bit mix
55// http://www.cris.com/~Ttwang/tech/inthash.htm
56PX_FORCE_INLINE uint32_t PxComputeHash(const uint32_t key)
57{
58 uint32_t k = key;
59 k += ~(k << 15);
60 k ^= (k >> 10);
61 k += (k << 3);
62 k ^= (k >> 6);
63 k += ~(k << 11);
64 k ^= (k >> 16);
65 return uint32_t(k);
66}
67
68PX_FORCE_INLINE uint32_t PxComputeHash(const int32_t key)
69{
70 return PxComputeHash(uint32_t(key));
71}
72
73// Thomas Wang's 64 bit mix
74// http://www.cris.com/~Ttwang/tech/inthash.htm
75PX_FORCE_INLINE uint32_t PxComputeHash(const uint64_t key)
76{
77 uint64_t k = key;
78 k += ~(k << 32);
79 k ^= (k >> 22);
80 k += ~(k << 13);
81 k ^= (k >> 8);
82 k += (k << 3);
83 k ^= (k >> 15);
84 k += ~(k << 27);
85 k ^= (k >> 31);
86 return uint32_t(UINT32_MAX & k);
87}

101// Hash function for pointers
102PX_INLINE uint32_t PxComputeHash(const void* ptr)
103{
104#if PX_P64_FAMILY
105 return PxComputeHash(uint64_t(ptr));
106#else
107 return PxComputeHash(uint32_t(UINT32_MAX & size_t(ptr)));
108#endif
109}

111// Hash function for pairs
112template <typename F, typename S>
113PX_INLINE uint32_t PxComputeHash(const PxPair<F, S>& p)
114{
115 uint32_t seed = 0x876543;
116 uint32_t m = 1000007;
117 return PxComputeHash(p.second) ^ (m * (PxComputeHash(p.first) ^ (m * seed)));
118}

120// hash object for hash map template parameter
121template <class Key>
122struct PxHash
123{
124 uint32_t operator()(const Key& k) const
125 {
126 return PxComputeHash(k);
127 }
128 bool equal(const Key& k0, const Key& k1) const
129 {
130 return k0 == k1;
131 }
132};

134// specialization for strings
135template <>
136struct PxHash<const char*>
137{
138 public:
139 uint32_t operator()(const char* _string) const
140 {
141 // "DJB" string hash
142 const uint8_t* string = reinterpret_cast<const uint8_t*>(_string);
143 uint32_t h = 5381;
144 for(const uint8_t* ptr = string; *ptr; ptr++)
145 h = ((h << 5) + h) ^ uint32_t(*ptr);
146 return h;
147 }
148 bool equal(const char* string0, const char* string1) const
149 {
150 return !Pxstrcmp(string0, string1);
151 }
152};
