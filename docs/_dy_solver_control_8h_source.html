63class BatchIterator
64{
65public:
66 PxConstraintBatchHeader* constraintBatchHeaders;
67 PxU32 mSize;
68 PxU32 mCurrentIndex;
69
70 BatchIterator(PxConstraintBatchHeader* _constraintBatchHeaders, PxU32 size) : constraintBatchHeaders(_constraintBatchHeaders),
71 mSize(size), mCurrentIndex(0)
72 {
73 }
74
75 PX_FORCE_INLINE const PxConstraintBatchHeader& GetCurrentHeader(const PxU32 constraintIndex)
76 {
77 PxU32 currentIndex = mCurrentIndex;
78 while((constraintIndex - constraintBatchHeaders[currentIndex].startIndex) >= constraintBatchHeaders[currentIndex].stride)
79 currentIndex = (currentIndex + 1)%mSize;
80 PxPrefetchLine(&constraintBatchHeaders[currentIndex], 128);
81 mCurrentIndex = currentIndex;
82 return constraintBatchHeaders[currentIndex];
83 }
84private:
85 BatchIterator& operator=(const BatchIterator&);
86};

89inline void SolveBlockParallel (PxSolverConstraintDesc* PX_RESTRICT constraintList, const PxI32 batchCount, const PxI32 index,
90 const PxI32 headerCount, SolverContext& cache, BatchIterator& iterator,
91 SolveBlockMethod solveTable[],
92 const PxI32 iteration
93 )
94{
95 const PxI32 indA = index - (iteration * headerCount);
96
97 const PxConstraintBatchHeader* PX_RESTRICT headers = iterator.constraintBatchHeaders;
98
99 const PxI32 endIndex = indA + batchCount;
100 for(PxI32 i = indA; i < endIndex; ++i)
101 {
102 const PxConstraintBatchHeader& header = headers[i];
103
104 const PxI32 numToGrab = header.stride;
105 PxSolverConstraintDesc* PX_RESTRICT block = &constraintList[header.startIndex];
106
107 PxPrefetch(block[0].constraint, 384);
108
109 for(PxI32 b = 0; b < numToGrab; ++b)
110 {
111 PxPrefetchLine(block[b].bodyA);
112 PxPrefetchLine(block[b].bodyB);
113 }
114
115 //OK. We have a number of constraints to run...
116 solveTable[header.constraintType](block, PxU32(numToGrab), cache);
117 }
118}
