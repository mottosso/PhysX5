61PX_FORCE_INLINE PxQuat PxVehicleComputeWheelLocalOrientation
62(const PxVehicleFrame& frame,
63 const PxVehicleSuspensionParams& suspensionParams,
64 const PxReal camberAngle, const PxReal toeAngle, const PxReal steerAngle,
65 const PxReal rotationAngle)
66{
67 const PxQuat wheelLocalOrientation =
68 (suspensionParams.suspensionAttachment.q * PxVehicleComputeRotation(frame, camberAngle, 0.0f, steerAngle + toeAngle))*

107PX_FORCE_INLINE PxTransform PxVehicleComputeWheelLocalPose
108(const PxVehicleFrame& frame,
109 const PxVehicleSuspensionParams& suspensionParams,
110 const PxVehicleSuspensionState& suspensionState,
111 const PxReal camberAngle, const PxReal toeAngle, const PxReal steerAngle,
112 const PxReal rotationAngle)
113{
114 //Full equation:
115 //PxTransform(suspAttachment.p + suspParams.suspensionTravelDir*suspDist, suspAttachment.q) *
116 //PxTransform(PxVec3(0), PxQuat(camber, 0, steer+toe)) *
117 //wheelAttachment *
118 //PxTransform(PxVec3(0), PxQuat(0, rotation, 0))
119 //Reduces to:
120 //PxTransform(suspAttachment.p + suspParams.suspensionTravelDir*suspDist, suspAttachment.q * PxQuat(camber, 0, steer+toe)) *
121 //PxTranfsorm(wheelAttachment.p, wheelAttachment.q * PxQuat(0, rotation, 0))
122 const PxF32 suspDist = (suspensionState.jounce != PX_VEHICLE_UNSPECIFIED_JOUNCE) ? (suspensionParams.suspensionTravelDist - suspensionState.jounce) : 0.0f;
123 const PxTransform wheelLocalPose =
124 PxTransform(
125 suspensionParams.suspensionAttachment.p + suspensionParams.suspensionTravelDir*suspDist,
126 suspensionParams.suspensionAttachment.q*PxVehicleComputeRotation(frame, camberAngle, 0.0f, steerAngle + toeAngle))*
127 PxTransform(
128 suspensionParams.wheelAttachment.p,
129 suspensionParams.wheelAttachment.q*PxVehicleComputeRotation(frame, 0.0f, rotationAngle, 0.0f));
130 return wheelLocalPose;
131}
