212 PX_INLINE void readName(const char*& name)
213 {
214 PxU32 len = *reinterpret_cast<PxU32*>(mExtraDataAddress);
215 mExtraDataAddress += sizeof(len);
216 name = len ? reinterpret_cast<const char*>(mExtraDataAddress) : NULL;
217 mExtraDataAddress += len;
218 }
219
227 template<typename T>
228 PX_INLINE T* readExtraData(PxU32 count=1)
229 {
230 T* data = reinterpret_cast<T*>(mExtraDataAddress);
231 mExtraDataAddress += sizeof(T)*count;
232 return data;
233 }
234
242 template<typename T, PxU32 alignment>
243 PX_INLINE T* readExtraData(PxU32 count=1)
244 {
245 alignExtraData(alignment);
246 return readExtraData<T>(count);
247 }
248
256 PX_INLINE void alignExtraData(PxU32 alignment = PX_SERIAL_ALIGN)
257 {
258 size_t addr = size_t(mExtraDataAddress);
259 addr = (addr+alignment-1)&~size_t(alignment-1);
260 mExtraDataAddress = reinterpret_cast<PxU8*>(addr);
261 }
262
263protected:
264
265 PxDeserializationContext() {}
266 virtual ~PxDeserializationContext() {}
267
268 PxU8* mExtraDataAddress;
269};
