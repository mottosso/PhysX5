106 template<typename T> PxU32 getConnectors(NpConnectorType::Enum type, T** userBuffer, PxU32 bufferSize, PxU32 startIndex=0) const
107 {
108 PxU32 nbConnectors = 0;
109 if(mConnectorArray)
110 {
111 for(PxU32 i=0; i<mConnectorArray->size(); i++)
112 {
113 NpConnector& c = (*mConnectorArray)[i];
114 if(c.mType == type && nbConnectors < bufferSize && i>=startIndex)
115 userBuffer[nbConnectors++] = static_cast<T*>(c.mObject);
116 }
117 }
118 return nbConnectors;
119 }
