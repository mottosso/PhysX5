231 {
232 PX_ASSERT(mRefCount == 0);
233 mRefCount = 1;
234 mCont = c;

239
247 PX_INLINE void setContinuation(PxBaseTask* c)
248 {
249 PX_ASSERT(c);
250 PX_ASSERT(mRefCount == 0);
251 mRefCount = 1;
252 mCont = c;
253 if(mCont)
254 {
255 mCont->addReference();
256 mTm = mCont->getTaskManager();
257 PX_ASSERT(mTm);
258 }
259 }

267 }

288 virtual void addReference() PX_OVERRIDE
289 {
290 mTm->addReference(*this);
291 }
292
298 virtual void release() PX_OVERRIDE

300 if(mCont)

302 }
