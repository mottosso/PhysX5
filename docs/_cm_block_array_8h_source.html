45template <typename T, PxU32 SlabSize = 4096>
46class BlockArray
47{
48 PxArray<T*> mBlocks;
49 PxU32 mSize;
50 PxU32 mCapacity;
51
52public:
53
54 BlockArray() : mSize(0), mCapacity(0)
55 {
56 }
57
58 ~BlockArray()
59 {
60 for (PxU32 a = 0; a < mBlocks.size(); ++a)
61 {
62 for (PxU32 i = 0; i < SlabSize; ++i)
63 {
64 mBlocks[a][i].~T();
65 }
66 PX_FREE(mBlocks[a]);
67 }
68 mBlocks.resize(0);
69 }
70
71 void reserve(PxU32 capacity)
72 {
73 if (capacity > mCapacity)
74 {
75 PxU32 nbSlabsRequired = (capacity + SlabSize - 1) / SlabSize;
76
77 PxU32 nbSlabsToAllocate = nbSlabsRequired - mBlocks.size();
78
79 mCapacity += nbSlabsToAllocate * SlabSize;
80
81 for (PxU32 a = 0; a < nbSlabsToAllocate; ++a)
82 {
83 T* ts = reinterpret_cast<T*>(PX_ALLOC(sizeof(T) * SlabSize, "BlockArray"));
84 for(PxU32 i = 0; i < SlabSize; ++i)
85 PX_PLACEMENT_NEW(ts+i, T)();
86 mBlocks.pushBack(ts);
87 }
88 }
89 }
90
91 void resize(PxU32 size)
92 {
93 reserve(size);
94 for (PxU32 a = mSize; a < size; ++a)
95 {
96 mBlocks[a / SlabSize][a&(SlabSize - 1)].~T();
97 mBlocks[a / SlabSize][a&(SlabSize-1)] = T();
98 }
99 mSize = size;
100 }
101
102 void forceSize_Unsafe(PxU32 size)
103 {
104 PX_ASSERT(size <= mCapacity);
105 mSize = size;
106 }
107
108 void remove(PxU32 idx)
109 {
110 PX_ASSERT(idx < mSize);
111 for (PxU32 a = idx; a < mSize; ++a)
112 {
113 mBlocks[a / SlabSize][a&(SlabSize-1)] = mBlocks[(a + 1) / SlabSize][(a + 1) &(SlabSize-1)];
114 }
115
116 mSize--;
117 mBlocks[mSize / SlabSize][mSize&(SlabSize - 1)].~T();
118 }
119
120 void replaceWithLast(PxU32 idx)
121 {
122 PX_ASSERT(idx < mSize);
123 --mSize;
124 mBlocks[idx / SlabSize][idx%SlabSize] = mBlocks[mSize / SlabSize][mSize%SlabSize];
125 }
126
127 T& operator [] (const PxU32 idx)
128 {
129 PX_ASSERT(idx < mSize);
130
131 return mBlocks[idx / SlabSize][idx%SlabSize];
132 }
133
134 const T& operator [] (const PxU32 idx) const
135 {
136 PX_ASSERT(idx < mSize);
137
138 return mBlocks[idx / SlabSize][idx%SlabSize];
139 }
140
141 void pushBack(const T& item)
142 {
143 reserve(mSize + 1);
144 mBlocks[mSize / SlabSize][mSize%SlabSize] = item;
145 mSize++;
146 }
147
148 PxU32 capacity() const { return mCapacity; }
149
150 PxU32 size() const { return mSize; }
151};
152
