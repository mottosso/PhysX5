124 PX_FORCE_INLINE void storeAny(const PxGeometry& geometry)
125 {
126 PX_ASSERT_WITH_MESSAGE( (geometry.getType() >= PxGeometryType::eSPHERE) &&
127 (geometry.getType() < PxGeometryType::eGEOMETRY_COUNT),
128 "Unexpected GeometryType in PxGeometryHolder::storeAny");
129
130 switch(geometry.getType())
131 {
132 case PxGeometryType::eSPHERE: put<PxSphereGeometry>(geometry); break;
133 case PxGeometryType::ePLANE: put<PxPlaneGeometry>(geometry); break;
134 case PxGeometryType::eCAPSULE: put<PxCapsuleGeometry>(geometry); break;
135 case PxGeometryType::eBOX: put<PxBoxGeometry>(geometry); break;
136 case PxGeometryType::eCONVEXMESH: put<PxConvexMeshGeometry>(geometry); break;
137 case PxGeometryType::eTRIANGLEMESH: put<PxTriangleMeshGeometry>(geometry); break;
138 case PxGeometryType::eTETRAHEDRONMESH: put<PxTetrahedronMeshGeometry>(geometry); break;
139 case PxGeometryType::eHEIGHTFIELD: put<PxHeightFieldGeometry>(geometry); break;
140 case PxGeometryType::ePARTICLESYSTEM: put<PxParticleSystemGeometry>(geometry); break;
141 case PxGeometryType::eHAIRSYSTEM: put<PxHairSystemGeometry>(geometry); break;
142 case PxGeometryType::eCUSTOM: put<PxCustomGeometry>(geometry); break;
143 case PxGeometryType::eGEOMETRY_COUNT:
144 case PxGeometryType::eINVALID: break;
145 }
146 }
