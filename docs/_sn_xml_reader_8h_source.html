81 //and sets the current node to inNode.

84 //the either way goto that child.
85 virtual void addOrGotoChild( const char* name ) = 0;
86 //Value is copied into the collection, inValue has no further references
87 //to it.
88 virtual void setCurrentItemValue( const char* value ) = 0;
89 //Removes the child but does not release the char* name or char* data ptrs.
90 //Those pointers are never released and are shared among collections.
91 //Thus copying nodes is cheap and safe.
92 virtual bool removeChild( const char* name ) = 0;
93 virtual void release() = 0;
94
95 bool renameProperty( const char* oldName, const char* newName )
96 {
97 if ( gotoChild( oldName ) )
98 {
99 const char* value = getCurrentItemValue();
100 leaveChild();
101 removeChild( oldName );
102 addOrGotoChild( newName );
103 setCurrentItemValue( value );
104 leaveChild();
105 return true;
106 }
107 return false;
108 }
109 bool readAndRemoveProperty( const char* name, const char*& outValue )
110 {
111 bool retval = read( name, outValue );
112 if ( retval ) removeChild( name );
113 return retval;
114 }
115
116 bool writePropertyIfNotEmpty( const char* name, const char* value )
117 {
118 if ( value && *value )
119 {
120 addOrGotoChild( name );
121 setCurrentItemValue( value );
122 leaveChild();
123 return true;
124 }
125 return false;
126 }
127 };
