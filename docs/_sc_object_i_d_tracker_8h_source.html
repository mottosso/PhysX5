48 PX_INLINE void releaseID(PxU32 id)

50 markIDAsDeleted(id);

53 PX_INLINE PxIntBool isDeletedID(PxU32 id) const { return mDeletedIDsMap.boundedTest(id); }

58 mDeletedIDsMap.resize(id);
59 mPendingReleasedIDs.reserve(numIds);
60 }
61 PX_INLINE void processPendingReleases()
62 {
63 for(PxU32 i=0; i < mPendingReleasedIDs.size(); i++)
64 {
65 mIDPool.freeID(mPendingReleasedIDs[i]);
66 }
67 mPendingReleasedIDs.clear();
68 }
69 PX_INLINE void reset()
70 {
71 processPendingReleases();
72 mPendingReleasedIDs.reset();
73
74 // Don't free stuff in IDPool, we still need the list of free IDs
75
76 // And it does not seem worth freeing the memory of the bitmap
77 }
78
79 PX_INLINE PxU32 getMaxID()
80 {
81 return mIDPool.getMaxID();
82 }
83 private:
84 PX_INLINE void markIDAsDeleted(PxU32 id) { PX_ASSERT(!isDeletedID(id)); mDeletedIDsMap.growAndSet(id); }
85
