132 // copy the hull from source
133 void copyHull(const ConvexHull& src)
134 {
135 mVertices.resize(src.getVertices().size());
136 mEdges.resize(src.getEdges().size());
137 mFacets.resize(src.getFacets().size());
138
139 PxMemCopy(mVertices.begin(), src.getVertices().begin(), src.getVertices().size()*sizeof(PxVec3));
140 PxMemCopy(mEdges.begin(), src.getEdges().begin(), src.getEdges().size()*sizeof(HalfEdge));
141 PxMemCopy(mFacets.begin(), src.getFacets().begin(), src.getFacets().size()*sizeof(PxPlane));
142 }
143
144 private:
145 PxArray<PxVec3> mVertices;
146 PxArray<HalfEdge> mEdges;

157 // three planes intersection
158 PX_FORCE_INLINE PxVec3 threePlaneIntersection(const PxPlane& p0, const PxPlane& p1, const PxPlane& p2)
159 {
160 PxMat33 mp = (PxMat33(p0.n, p1.n, p2.n)).getTranspose();
161 PxMat33 mi = (mp).getInverse();
162 PxVec3 b(p0.d, p1.d, p2.d);
163 return -mi.transform(b);
164 }
