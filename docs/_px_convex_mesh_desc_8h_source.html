269PX_INLINE bool PxConvexMeshDesc::isValid() const
270{
271 // Check geometry
272 if(points.count < 3 || //at least 1 trig's worth of points
273 (points.count > 0xffff && flags & PxConvexFlag::e16_BIT_INDICES))
274 return false;
275 if(!points.data)
276 return false;
277 if(points.stride < sizeof(PxVec3)) //should be at least one point's worth of data
278 return false;
279 if (quantizedCount < 4)
280 return false;
281
282 // Check topology
283 if(polygons.data)
284 {
285 if(polygons.count < 4) // we require 2 neighbors for each vertex - 4 polygons at least
286 return false;
287
288 if(!indices.data) // indices must be provided together with polygons
289 return false;
290
291 PxU32 limit = (flags & PxConvexFlag::e16_BIT_INDICES) ? sizeof(PxU16) : sizeof(PxU32);
292 if(indices.stride < limit)
293 return false;
294
295 limit = sizeof(PxHullPolygon);
296 if(polygons.stride < limit)
297 return false;
298 }
299 else
300 {
301 // We can compute the hull from the vertices
302 if(!(flags & PxConvexFlag::eCOMPUTE_CONVEX))
303 return false; // If the mesh is convex and we're not allowed to compute the hull,
304 // you have to provide it completely (geometry & topology).
305 }
306
307 if((flags & PxConvexFlag::ePLANE_SHIFTING) && vertexLimit < 4)
308 {
309 return false;
310 }
311
312 if (!(flags & PxConvexFlag::ePLANE_SHIFTING) && vertexLimit < 8)
313 {
314 return false;
315 }
316
317 if(vertexLimit > 255)
318 {
319 return false;
320 }
321
322 if (polygonLimit < 4)
323 {
324 return false;
325 }
326
327 if (polygonLimit > 255)
328 {
329 return false;
330 }
331
332 if (sdfDesc && !sdfDesc->isValid())
333 {
334 return false;
335 }
336
337 return true;
338}
