44class SpInertia
45{
46public:
47 SpInertia() {}
48
49 SpInertia(const PxMat33& ll, const PxMat33& la, const PxMat33& aa): mLL(ll), mLA(la), mAA(aa)
50 {
51 }
52
53 static SpInertia getZero()
54 {
55 return SpInertia(PxMat33(PxZero), PxMat33(PxZero),
56 PxMat33(PxZero));
57 }
58
59 static SpInertia dyad(const Cm::SpatialVector& column, const Cm::SpatialVector& row)
60 {
61 return SpInertia(dyad(column.linear, row.linear),
62 dyad(column.linear, row.angular),
63 dyad(column.angular, row.angular));
64 }
65
66
67 static SpInertia inertia(PxReal mass, const PxVec3& inertia)
68 {
69 return SpInertia(PxMat33::createDiagonal(PxVec3(mass,mass,mass)), PxMat33(PxZero),
70 PxMat33::createDiagonal(inertia));
71 }
72
73
74 SpInertia operator+(const SpInertia& m) const
75 {
76 return SpInertia(mLL+m.mLL, mLA+m.mLA, mAA+m.mAA);
77 }
78
79 SpInertia operator-(const SpInertia& m) const
80 {
81 return SpInertia(mLL-m.mLL, mLA-m.mLA, mAA-m.mAA);
82 }
83
84 SpInertia operator*(PxReal r) const
85 {
86 return SpInertia(mLL*r, mLA*r, mAA*r);
87 }
88
89 void operator+=(const SpInertia& m)
90 {
91 mLL+=m.mLL;
92 mLA+=m.mLA;
93 mAA+=m.mAA;
94 }
95
96 void operator-=(const SpInertia& m)
97 {
98 mLL-=m.mLL;
99 mLA-=m.mLA;
100 mAA-=m.mAA;
101 }
102
103
104 PX_FORCE_INLINE Cm::SpatialVector operator *(const Cm::SpatialVector& v) const
105 {
106 return Cm::SpatialVector(mLL*v.linear +mLA*v.angular,
107 mLA.transformTranspose(v.linear)+mAA*v.angular);
108 }
109
110 SpInertia operator *(const SpInertia& v) const
111 {
112 return SpInertia(mLL*v.mLL + mLA * v.mLA.getTranspose(),
113 mLL*v.mLA + mLA * v.mAA,
114 mLA.getTranspose()*v.mLA + mAA * v.mAA);
115 }
116
117
118 bool isFinite() const
119 {
120 return true;
121// return mLL.isFinite() && mLA.isFinite() && mAA.isFinite();
122 }
123
124 PxMat33 mLL, mLA; // linear force from angular motion, linear force from linear motion
125 PxMat33 mAA; // angular force from angular motion, mAL = mLA.transpose()
126
127private:
128 static PxMat33 dyad(PxVec3 col, PxVec3 row)
129 {
130 return PxMat33(col*row.x, col*row.y, col*row.z);
131 }
132
