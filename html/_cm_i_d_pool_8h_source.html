40 template<class FreeBuffer>
41 class IDPoolBase : public PxUserAllocated
42 {
43 protected:
44 PxU32 mCurrentID;
45 FreeBuffer mFreeIDs;
46 public:
47 IDPoolBase() : mCurrentID(0) {}
48
49 void freeID(PxU32 id)
50 {
51 // Allocate on first call
52 // Add released ID to the array of free IDs
53 if(id == (mCurrentID - 1))
54 --mCurrentID;
55 else
56 mFreeIDs.pushBack(id);
57 }
58
59 void freeAll()
60 {
61 mCurrentID = 0;
62 mFreeIDs.clear();
63 }
64
65 PxU32 getNewID()
66 {
67 // If recycled IDs are available, use them
68 const PxU32 size = mFreeIDs.size();
69 if(size)
70 {
71 // Recycle last ID
72 return mFreeIDs.popBack();
73 }
74 // Else create a new ID
75 return mCurrentID++;
76 }
77
78 PxU32 getNumUsedID() const
79 {
80 return mCurrentID - mFreeIDs.size();
81 }
82
83 PxU32 getMaxID() const
84 {
85 return mCurrentID;
86 }
87
88 };

90 //This class extends IDPoolBase. This is mainly used for when it is unsafe for the application to free the id immediately so that it can
91 //defer the free process until it is safe to do so
92 template<class FreeBuffer>
93 class DeferredIDPoolBase : public IDPoolBase<FreeBuffer>
94 {
95 FreeBuffer mDeferredFreeIDs;
96 public:
97 //release an index into the deferred list
98 void deferredFreeID(PxU32 id)
99 {
100 mDeferredFreeIDs.pushBack(id);
101 }
102
103 //release the deferred indices into the free list
104 void processDeferredIds()
105 {
106 const PxU32 deferredFreeIDCount = mDeferredFreeIDs.size();
107 for(PxU32 a = 0; a < deferredFreeIDCount;++a)
108 {
109 IDPoolBase<FreeBuffer>::freeID(mDeferredFreeIDs[a]);
110 }
111 mDeferredFreeIDs.clear();
112 }
113
114 //release all indices
115 void freeAll()
116 {
117 mDeferredFreeIDs.clear();
118 IDPoolBase<FreeBuffer>::freeAll();
119 }
120
121 PxU32 getNumUsedID() const
122 {
123 return IDPoolBase<FreeBuffer>::getNumUsedID() - mDeferredFreeIDs.size();
124 }
125
126 FreeBuffer& getDeferredFreeIDs() { return mDeferredFreeIDs; }
127 };

129 //This is spu friendly fixed size array
130 template <typename T, uint32_t N>
131 class InlineFixedArray
132 {
133 T mArr[N];
134 PxU32 mSize;
135 public:
136
137 InlineFixedArray() : mSize(0)
138 {
139 }
140
141 ~InlineFixedArray(){}
142
143 void pushBack(const T& t)
144 {
145 PX_ASSERT(mSize < N);
146 mArr[mSize++] = t;
147 }
148
149 T popBack()
150 {
151 PX_ASSERT(mSize > 0);
152 return mArr[--mSize];
153 }
154
155 void clear() { mSize = 0; }
156
157 T& operator [] (PxU32 index) { PX_ASSERT(index < N); return mArr[index]; }
158
159 const T& operator [] (PxU32 index) const { PX_ASSERT(index < N); return mArr[index]; }
160
161 PxU32 size() const { return mSize; }
162 };
