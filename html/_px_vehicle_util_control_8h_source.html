51struct PX_DEPRECATED PxVehicleSteerFilter
52{
53 PxVehicleSteerFilter(float sharpness=0.0f) : mSharpness(sharpness), mFilteredMaxSteer(0.0f) {}
54
55 static PX_FORCE_INLINE float feedbackFilter(float val, float& memory, float sharpness)
56 {
57 if(sharpness<0.0f) sharpness = 0.0f;
58 else if(sharpness>1.0f) sharpness = 1.0f;
59 return memory = val * sharpness + memory * (1.0f - sharpness);
60 }
61
62 PX_FORCE_INLINE float computeMaxSteer(const bool isVehicleInAir, const PxFixedSizeLookupTable<8>& steerVsForwardSpeedTable,
63 const PxF32 vzAbs, const PxF32 timestep) const
64 {
65 const PxF32 targetMaxSteer = (isVehicleInAir ? 1.0f : steerVsForwardSpeedTable.getYVal(vzAbs));
66 if(mSharpness==0.0f)
67 return targetMaxSteer;
68 else
69 return feedbackFilter(targetMaxSteer, mFilteredMaxSteer, mSharpness*timestep);
70 }
71
72 PxReal mSharpness;
73 mutable PxReal mFilteredMaxSteer;
74};
