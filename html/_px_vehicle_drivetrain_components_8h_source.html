74class PxVehicleDirectDriveCommandResponseComponent : public PxVehicleComponent
75{
76public:
77
78 PxVehicleDirectDriveCommandResponseComponent() : PxVehicleComponent() {}
79 virtual ~PxVehicleDirectDriveCommandResponseComponent() {}
80
96 virtual void getDataForDirectDriveCommandResponseComponent(
97 const PxVehicleAxleDescription*& axleDescription,
98 PxVehicleSizedArrayData<const PxVehicleBrakeCommandResponseParams>& brakeResponseParams,
99 const PxVehicleDirectDriveThrottleCommandResponseParams*& throttleResponseParams,
100 const PxVehicleSteerCommandResponseParams*& steerResponseParams,
101 PxVehicleSizedArrayData<const PxVehicleAckermannParams>& ackermannParams,
102 const PxVehicleCommandState*& commands, const PxVehicleDirectDriveTransmissionCommandState*& transmissionCommands,
103 const PxVehicleRigidBodyState*& rigidBodyState,
104 PxVehicleArrayData<PxReal>& brakeResponseStates,
105 PxVehicleArrayData<PxReal>& throttleResponseStates,
106 PxVehicleArrayData<PxReal>& steerResponseStates) = 0;
107
112 virtual bool update(const PxReal dt, const PxVehicleSimulationContext& context)
113 {
114 PX_UNUSED(dt);
115 PX_UNUSED(context);
116
117 PX_PROFILE_ZONE("PxVehicleDirectDriveCommandResponseComponent::update", 0);
118
119 const PxVehicleAxleDescription* axleDescription;
120 PxVehicleSizedArrayData<const PxVehicleBrakeCommandResponseParams> brakeResponseParams;
121 const PxVehicleDirectDriveThrottleCommandResponseParams* throttleResponseParams;
122 const PxVehicleSteerCommandResponseParams* steerResponseParams;
123 PxVehicleSizedArrayData<const PxVehicleAckermannParams> ackermannParams;
124 const PxVehicleCommandState* commands;
125 const PxVehicleDirectDriveTransmissionCommandState* transmissionCommands;
126 const PxVehicleRigidBodyState* rigidBodyState;
127 PxVehicleArrayData<PxReal> brakeResponseStates;
128 PxVehicleArrayData<PxReal> throttleResponseStates;
129 PxVehicleArrayData<PxReal> steerResponseStates;
130
131 getDataForDirectDriveCommandResponseComponent(axleDescription,
132 brakeResponseParams, throttleResponseParams, steerResponseParams, ackermannParams,
133 commands, transmissionCommands,
134 rigidBodyState,
135 brakeResponseStates, throttleResponseStates, steerResponseStates);
136
137 const PxReal longitudinalSpeed = rigidBodyState->getLongitudinalSpeed(context.frame);
138
139 for (PxU32 i = 0; i < axleDescription->nbWheels; i++)
140 {
141 const PxU32 wheelId = axleDescription->wheelIdsInAxleOrder[i];
142
143 PxVehicleBrakeCommandResponseUpdate(
144 commands->brakes, commands->nbBrakes, longitudinalSpeed,
145 wheelId, brakeResponseParams,
146 brakeResponseStates[wheelId]);
147
148 PxVehicleDirectDriveThrottleCommandResponseUpdate(
149 commands->throttle, *transmissionCommands, longitudinalSpeed,
150 wheelId, *throttleResponseParams,
151 throttleResponseStates[wheelId]);
152
153 PxVehicleSteerCommandResponseUpdate(
154 commands->steer, longitudinalSpeed,
155 wheelId, *steerResponseParams,
156 steerResponseStates[wheelId]);
157 }
158 if (ackermannParams.size > 0)
159 PxVehicleAckermannSteerUpdate(
160 commands->steer,
161 *steerResponseParams, ackermannParams,
162 steerResponseStates);
163
164 return true;
165 }
166};

302class PxVehicleEngineDriveCommandResponseComponent : public PxVehicleComponent
303{
304public:
305
306 PxVehicleEngineDriveCommandResponseComponent() : PxVehicleComponent() {}
307 virtual ~PxVehicleEngineDriveCommandResponseComponent() {}
308
335 virtual void getDataForEngineDriveCommandResponseComponent(
336 const PxVehicleAxleDescription*& axleDescription,
337 PxVehicleSizedArrayData<const PxVehicleBrakeCommandResponseParams>& brakeResponseParams,
338 const PxVehicleSteerCommandResponseParams*& steerResponseParams,
339 PxVehicleSizedArrayData<const PxVehicleAckermannParams>& ackermannParams,
340 const PxVehicleGearboxParams*& gearboxParams,
341 const PxVehicleClutchCommandResponseParams*& clutchResponseParams,
342 const PxVehicleEngineParams*& engineParams,
343 const PxVehicleRigidBodyState*& rigidBodyState,
344 const PxVehicleEngineState*& engineState,
345 const PxVehicleAutoboxParams*& autoboxParams,
346 const PxVehicleCommandState*& commands,
347 const PxVehicleEngineDriveTransmissionCommandState*& transmissionCommands,
348 PxVehicleArrayData<PxReal>& brakeResponseStates,
349 PxVehicleEngineDriveThrottleCommandResponseState*& throttleResponseState,
350 PxVehicleArrayData<PxReal>& steerResponseStates,
351 PxVehicleGearboxState*& gearboxResponseState,
352 PxVehicleClutchCommandResponseState*& clutchResponseState,
353 PxVehicleAutoboxState*& autoboxState) = 0;
354
355 virtual bool update(const PxReal dt, const PxVehicleSimulationContext& context)
356 {
357 PX_UNUSED(dt);
358 PX_UNUSED(context);
359
360 PX_PROFILE_ZONE("PxVehicleEngineDriveCommandResponseComponent::update", 0);
361
362 const PxVehicleAxleDescription* axleDescription;
363 PxVehicleSizedArrayData<const PxVehicleBrakeCommandResponseParams> brakeResponseParams;
364 const PxVehicleSteerCommandResponseParams* steerResponseParams;
365 PxVehicleSizedArrayData<const PxVehicleAckermannParams> ackermannParams;
366 const PxVehicleGearboxParams* gearboxParams;
367 const PxVehicleClutchCommandResponseParams* clutchResponseParams;
368 const PxVehicleEngineParams* engineParams;
369 const PxVehicleRigidBodyState* rigidBodyState;
370 const PxVehicleEngineState* engineState;
371 const PxVehicleAutoboxParams* autoboxParams;
372 const PxVehicleCommandState* commands;
373 const PxVehicleEngineDriveTransmissionCommandState* transmissionCommands;
374 PxVehicleArrayData<PxReal> brakeResponseStates;
375 PxVehicleEngineDriveThrottleCommandResponseState* throttleResponseState;
376 PxVehicleArrayData<PxReal> steerResponseStates;
377 PxVehicleGearboxState* gearboxResponseState;
378 PxVehicleClutchCommandResponseState* clutchResponseState;
379 PxVehicleAutoboxState* autoboxState;
380
381 getDataForEngineDriveCommandResponseComponent(axleDescription,
382 brakeResponseParams, steerResponseParams, ackermannParams,
383 gearboxParams, clutchResponseParams, engineParams, rigidBodyState,
384 engineState, autoboxParams,
385 commands, transmissionCommands,
386 brakeResponseStates, throttleResponseState,
387 steerResponseStates,
388 gearboxResponseState, clutchResponseState, autoboxState);
389
390 //The autobox can modify commands like throttle and target gear. Since the user defined
391 //values should not be overwritten, a copy is used to compute the response.
392 PxVehicleCommandState commandsTmp = *commands;
393 PxVehicleEngineDriveTransmissionCommandState transmissionCommandsTmp = *transmissionCommands;
394
395 const PxReal longitudinalSpeed = rigidBodyState->getLongitudinalSpeed(context.frame);
396
397 //Let the autobox set the target gear, unless the user defined target gear requests
398 //a shift already
399 if (autoboxParams)
400 {
401 PX_ASSERT(engineParams);
402 PX_ASSERT(engineState);
403 PX_ASSERT(autoboxState);
404
405 PxVehicleAutoBoxUpdate(
406 *engineParams, *gearboxParams, *autoboxParams,
407 *engineState, *gearboxResponseState, dt,
408 transmissionCommandsTmp.targetGear, *autoboxState, commandsTmp.throttle);
409 }
410 else if (transmissionCommandsTmp.targetGear == PxVehicleEngineDriveTransmissionCommandState::eAUTOMATIC_GEAR)
411 {
412 //If there is no autobox but eAUTOMATIC_GEAR was specified, use the current target gear
413 transmissionCommandsTmp.targetGear = gearboxResponseState->targetGear;
414 }
415
416 //Distribute brake torque to the wheels across each axle.
417 for (PxU32 i = 0; i < axleDescription->nbWheels; i++)
418 {
419 const PxU32 wheelId = axleDescription->wheelIdsInAxleOrder[i];
420 PxVehicleBrakeCommandResponseUpdate(
421 commandsTmp.brakes, commandsTmp.nbBrakes, longitudinalSpeed,
422 wheelId, brakeResponseParams,
423 brakeResponseStates[i]);
424 }
425
426 //Update target gear as required.
427 PxVehicleGearCommandResponseUpdate(
428 transmissionCommandsTmp.targetGear,
429 *gearboxParams,
430 *gearboxResponseState);
431
432 //Compute the response to the clutch command.
433 PxVehicleClutchCommandResponseLinearUpdate(
434 transmissionCommandsTmp.clutch,
435 *clutchResponseParams,
436 *clutchResponseState);
437
438 //Compute the response to the throttle command.
439 PxVehicleEngineDriveThrottleCommandResponseLinearUpdate(
440 commandsTmp,
441 *throttleResponseState);
442
443 //Update the steer angles and Ackermann correction.
444 for (PxU32 i = 0; i < axleDescription->nbWheels; i++)
445 {
446 const PxU32 wheelId = axleDescription->wheelIdsInAxleOrder[i];
447 PxVehicleSteerCommandResponseUpdate(commandsTmp.steer, longitudinalSpeed, wheelId, *steerResponseParams, steerResponseStates[i]);
448 }
449 if (ackermannParams.size > 0)
450 PxVehicleAckermannSteerUpdate(commandsTmp.steer, *steerResponseParams, ackermannParams, steerResponseStates);
451
452 return true;
453 }
454};
