63 // hashing function for PrunerPayload key
64 // PT: TODO: move this to PrunerPayload?
65 struct ExtendedBucketPrunerHash
66 {
67 PX_FORCE_INLINE uint32_t operator()(const PrunerPayload& payload) const
68 {
69#if PX_P64_FAMILY
70// const PxU32 h0 = PxHash((const void*)payload.data[0]);
71// const PxU32 h1 = PxHash((const void*)payload.data[1]);
72 const PxU32 h0 = PxU32(PX_MAX_U32 & payload.data[0]);
73 const PxU32 h1 = PxU32(PX_MAX_U32 & payload.data[1]);
74 return physx::PxComputeHash(PxU64(h0) | (PxU64(h1) << 32));
75#else
76 return physx::PxComputeHash(PxU64(payload.data[0]) | (PxU64(payload.data[1]) << 32));
77#endif
78 }
79 PX_FORCE_INLINE bool equal(const PrunerPayload& k0, const PrunerPayload& k1) const
80 {
81 return (k0.data[0] == k1.data[0]) && (k0.data[1] == k1.data[1]);
82 }
83 };
