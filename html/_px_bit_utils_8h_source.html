41PX_INLINE uint32_t PxBitCount(uint32_t v)
42{
43 // from http://graphics.stanford.edu/~seander/bithacks.html#CountBitsSetParallel
44 uint32_t const w = v - ((v >> 1) & 0x55555555);
45 uint32_t const x = (w & 0x33333333) + ((w >> 2) & 0x33333333);
46 return (((x + (x >> 4)) & 0xF0F0F0F) * 0x1010101) >> 24;
47}

49PX_INLINE bool PxIsPowerOfTwo(uint32_t x)
50{
51 return x != 0 && (x & (x - 1)) == 0;

54// "Next Largest Power of 2
55// Given a binary integer value x, the next largest power of 2 can be computed by a SWAR algorithm
56// that recursively "folds" the upper bits into the lower bits. This process yields a bit vector with
57// the same most significant 1 as x, but all 1's below it. Adding 1 to that value yields the next
58// largest power of 2. For a 32-bit value:"
59PX_INLINE uint32_t PxNextPowerOfTwo(uint32_t x)
60{
61 x |= (x >> 1);
62 x |= (x >> 2);
63 x |= (x >> 4);
64 x |= (x >> 8);
65 x |= (x >> 16);
66 return x + 1;
67}

89// Helper function to approximate log2 of an integer value
90// assumes that the input is actually power of two.
91PX_INLINE uint32_t PxILog2(uint32_t num)
92{
93 for(uint32_t i = 0; i < 32; i++)
94 {
95 num >>= 1;
96 if(num == 0)
97 return i;
98 }
99
100 PX_ASSERT(0);
101 return uint32_t(-1);
102}
