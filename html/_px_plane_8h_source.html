83 PX_CUDA_CALLABLE PX_FORCE_INLINE PxPlane(const PxVec3& p0, const PxVec3& p1, const PxVec3& p2)
84 {
85 n = (p1 - p0).cross(p2 - p0).getNormalized();
86 d = -p0.dot(n);
87 }
88
92 PX_CUDA_CALLABLE PX_INLINE bool operator==(const PxPlane& p) const
93 {
94 return n == p.n && d == p.d;
95 }
96
97 PX_CUDA_CALLABLE PX_FORCE_INLINE float distance(const PxVec3& p) const
98 {
99 return p.dot(n) + d;
100 }
101
102 PX_CUDA_CALLABLE PX_FORCE_INLINE bool contains(const PxVec3& p) const
103 {
104 return PxAbs(distance(p)) < (1.0e-7f);
105 }
106
110 PX_CUDA_CALLABLE PX_FORCE_INLINE PxVec3 project(const PxVec3& p) const
111 {
112 return p - n * distance(p);
113 }
114
118 PX_CUDA_CALLABLE PX_FORCE_INLINE PxVec3 pointInPlane() const
119 {
120 return -n * d;
121 }
122
127 PX_CUDA_CALLABLE PX_FORCE_INLINE void normalize()
128 {
129 float denom = 1.0f / n.magnitude();
130 n *= denom;
131 d *= denom;
132 }
133
137 PX_CUDA_CALLABLE PX_FORCE_INLINE PxPlane transform(const PxTransform& pose) const
138 {
139 const PxVec3 transformedNormal = pose.rotate(n);
140 return PxPlane(transformedNormal, d - pose.p.dot(transformedNormal));
141 }
142
146 PX_CUDA_CALLABLE PX_FORCE_INLINE PxPlane inverseTransform(const PxTransform& pose) const
147 {
148 const PxVec3 transformedNormal = pose.rotateInv(n);
149 return PxPlane(transformedNormal, d + pose.p.dot(n));
150 }
151
152 PxVec3 n;
153 float d;
