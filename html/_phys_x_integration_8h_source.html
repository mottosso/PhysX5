61 PX_FORCE_INLINE bool isValid(const PxVehicleAxleDescription& axleDesc) const
62 {
63 if (!physxRoadGeometryQueryParams.isValid())
64 return false;
65 for (PxU32 i = 0; i < axleDesc.nbWheels; i++)
66 {
67 const PxU32 wheelId = axleDesc.wheelIdsInAxleOrder[i];
68 if (!physxMaterialFrictionParams[wheelId].isValid())
69 return false;
70 if (!physxSuspensionLimitConstraintParams[wheelId].isValid())
71 return false;
72 }
73 return true;
74 }
