100 // PT: computes proper impact data for sphere-sweep-vs-tri, after the closest tri has been found
101 PX_FORCE_INLINE bool computeSphereTriangleImpactData(PxGeomSweepHit& h, PxVec3& triNormalOut, PxU32 index, PxReal curT,
102 const PxVec3& center, const PxVec3& unitDir, const PxVec3& bestTriNormal,
103 const PxTriangle* PX_RESTRICT triangles,
104 bool isDoubleSided, bool meshBothSides)
105 {
106 if(index==PX_INVALID_U32)
107 return false; // We didn't touch any triangle
108
109 // Compute impact data only once, using best triangle
110 PxVec3 hitPos, normal;
111 computeSphereTriImpactData(hitPos, normal, center, unitDir, curT, triangles[index]);
112
113 // PT: by design, returned normal is opposed to the sweep direction.
114 if(shouldFlipNormal(normal, meshBothSides, isDoubleSided, bestTriNormal, unitDir))
115 normal = -normal;
116
117 h.position = hitPos;
118 h.normal = normal;
119 h.distance = curT;
120 h.faceIndex = index;
121 h.flags = PxHitFlag::eNORMAL|PxHitFlag::ePOSITION;
122 triNormalOut = bestTriNormal;
123 return true;
124 }
