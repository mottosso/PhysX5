124 virtual bool update(const PxReal dt, const PxVehicleSimulationContext& context)
125 {
126 PX_PROFILE_ZONE("PxVehicleSuspensionComponent::update", 0);
127
128 const PxVehicleAxleDescription* axleDescription;
129 const PxVehicleRigidBodyParams* rigidBodyParams;
130 const PxVehicleSuspensionStateCalculationParams* suspensionStateCalculationParams;
131 PxVehicleArrayData<const PxReal> steerResponseStates;
132 const PxVehicleRigidBodyState* rigidBodyState;
133 PxVehicleArrayData<const PxVehicleWheelParams> wheelParams;
134 PxVehicleArrayData<const PxVehicleSuspensionParams> suspensionParams;
135 PxVehicleArrayData<const PxVehicleSuspensionComplianceParams> suspensionComplianceParams;
136 PxVehicleArrayData<const PxVehicleSuspensionForceParams> suspensionForceParams;
137 PxVehicleSizedArrayData<const PxVehicleAntiRollForceParams> antiRollForceParams;
138 PxVehicleArrayData<const PxVehicleRoadGeometryState> wheelRoadGeomStates;
139 PxVehicleArrayData<PxVehicleSuspensionState> suspensionStates;
140 PxVehicleArrayData<PxVehicleSuspensionComplianceState> suspensionComplianceStates;
141 PxVehicleArrayData<PxVehicleSuspensionForce> suspensionForces;
142 PxVehicleAntiRollTorque* antiRollTorque;
143
144 getDataForSuspensionComponent(axleDescription, rigidBodyParams, suspensionStateCalculationParams,
145 steerResponseStates, rigidBodyState,
146 wheelParams, suspensionParams,
147 suspensionComplianceParams, suspensionForceParams, antiRollForceParams,
148 wheelRoadGeomStates, suspensionStates, suspensionComplianceStates,
149 suspensionForces, antiRollTorque);
150
151 for (PxU32 i = 0; i < axleDescription->nbWheels; i++)
152 {
153 const PxU32 wheelId = axleDescription->wheelIdsInAxleOrder[i];
154
155 //Update the suspension state (jounce, jounce speed)
156 PxVehicleSuspensionStateUpdate(
157 wheelParams[wheelId], suspensionParams[wheelId], *suspensionStateCalculationParams,
158 suspensionForceParams[wheelId].stiffness, suspensionForceParams[wheelId].damping,
159 steerResponseStates[wheelId], wheelRoadGeomStates[wheelId],
160 *rigidBodyState,
161 dt, context.frame, context.gravity,
162 suspensionStates[wheelId]);
163
164 //Update the compliance from the suspension state.
165 PxVehicleSuspensionComplianceUpdate(
166 suspensionParams[wheelId], suspensionComplianceParams[wheelId],
167 suspensionStates[wheelId],
168 suspensionComplianceStates[wheelId]);
169
170 //Compute the suspension force from the suspension and compliance states.
171 PxVehicleSuspensionForceUpdate(
172 suspensionParams[wheelId], suspensionForceParams[wheelId],
173 wheelRoadGeomStates[wheelId], suspensionStates[wheelId],
174 suspensionComplianceStates[wheelId], *rigidBodyState,
175 context.gravity, rigidBodyParams->mass,
176 suspensionForces[wheelId]);
177 }
178
179 if (antiRollForceParams.size>0 && antiRollTorque)
180 {
181 PxVehicleAntiRollForceUpdate(
182 suspensionParams, antiRollForceParams,
183 suspensionStates.getConst(), suspensionComplianceStates.getConst(), *rigidBodyState,
184 *antiRollTorque);
185 }
186
187 return true;
188 }
189};
