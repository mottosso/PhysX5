157class PxVehiclePhysXActorEndComponent : public PxVehicleComponent
158{
159public:
160
161 PxVehiclePhysXActorEndComponent() : PxVehicleComponent() {}
162 virtual ~PxVehiclePhysXActorEndComponent() {}
163
177 virtual void getDataForPhysXActorEndComponent(
178 const PxVehicleAxleDescription*& axleDescription,
179 const PxVehicleRigidBodyState*& rigidBodyState,
180 PxVehicleArrayData<const PxVehicleWheelParams>& wheelParams,
181 PxVehicleArrayData<const PxTransform>& wheelShapeLocalPoses,
182 PxVehicleArrayData<const PxVehicleWheelRigidBody1dState>& wheelRigidBody1dStates,
183 PxVehicleArrayData<const PxVehicleWheelLocalPose>& wheelLocalPoses,
184 const PxVehicleGearboxState*& gearState,
185 PxVehiclePhysXActor*& physxActor) = 0;
186
187 virtual bool update(const PxReal dt, const PxVehicleSimulationContext& context)
188 {
189 PX_UNUSED(dt);
190
191 PX_PROFILE_ZONE("PxVehiclePhysXActorEndComponent::update", 0);
192
193 const PxVehicleAxleDescription* axleDescription;
194 const PxVehicleRigidBodyState* rigidBodyState;
195 PxVehicleArrayData<const PxVehicleWheelParams> wheelParams;
196 PxVehicleArrayData<const PxTransform> wheelShapeLocalPoses;
197 PxVehicleArrayData<const PxVehicleWheelRigidBody1dState> wheelRigidBody1dStates;
198 PxVehicleArrayData<const PxVehicleWheelLocalPose> wheelLocalPoses;
199 const PxVehicleGearboxState* gearState;
200 PxVehiclePhysXActor* physxActor;
201
202 getDataForPhysXActorEndComponent(axleDescription, rigidBodyState,
203 wheelParams, wheelShapeLocalPoses, wheelRigidBody1dStates, wheelLocalPoses, gearState,
204 physxActor);
205
206 for (PxU32 i = 0; i < axleDescription->nbWheels; i++)
207 {
208 const PxU32 wheelId = axleDescription->wheelIdsInAxleOrder[i];
209
210 PxVehicleWriteWheelLocalPoseToPhysXWheelShape(wheelLocalPoses[wheelId].localPose, wheelShapeLocalPoses[wheelId],
211 physxActor->wheelShapes[wheelId]);
212 }
213
214 if (context.getType() == PxVehicleSimulationContextType::ePHYSX)
215 {
216 const PxVehiclePhysXSimulationContext& physxContext = static_cast<const PxVehiclePhysXSimulationContext&>(context);
217
218 PxVehicleWriteRigidBodyStateToPhysXActor(physxContext.physxActorUpdateMode, *rigidBodyState, dt, *physxActor->rigidBody);
219
220 PxVehiclePhysxActorKeepAwakeCheck(*axleDescription, wheelParams, wheelRigidBody1dStates,
221 physxContext.physxActorWakeCounterThreshold, physxContext.physxActorWakeCounterResetValue, gearState,
222 *physxActor->rigidBody);
223 }
224 else
225 {
226 PX_ALWAYS_ASSERT();
227 }
228
229 return true;
230 }
231};
