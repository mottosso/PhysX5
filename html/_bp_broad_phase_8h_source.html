75/*
76\brief Structure used to report created and deleted broadphase pairs
77\note The indices mVolA and mVolB correspond to the bounds indices
78BroadPhaseUpdateData::mCreated used by BroadPhase::update
79@see BroadPhase::getCreatedPairs, BroadPhase::getDeletedPairs
80*/
81struct BroadPhasePair
82{
83 BroadPhasePair(ShapeHandle volA, ShapeHandle volB) :
84 mVolA (PxMin(volA, volB)),
85 mVolB (PxMax(volA, volB))

87 }
88 BroadPhasePair() :
89 mVolA (BP_INVALID_BP_HANDLE),
90 mVolB (BP_INVALID_BP_HANDLE)
91 {
92 }
93
94 ShapeHandle mVolA; // NB: mVolA < mVolB
95 ShapeHandle mVolB;
96};

159 /*
160 \brief Get created pairs.
161
162 Note that each overlap pair is reported only on the frame when the overlap first occurs. The overlap persists
163 until the pair appears in the list of deleted pairs or either of the bounds in the pair is removed from the broadphase.
164 A created overlap must involve at least one of the bounds of the overlap pair appearing in either the created or updated list.
165 It is impossible for the same pair to appear simultaneously in the list of created and deleted overlap pairs.
166 An overlap is defined as a pair of bounds that overlap on all three axes; that is when maxA > minB and maxB > minA for all three axes.
167
168 \param nbCreatedPairs [out] The number of created aabb overlap pairs computed in the execution of update() that has just completed.
169 \return The array of created aabb overlap pairs computed in the execution of update() that has just completed.
170 */
