114PX_INLINE bool PxBVHDesc::isValid() const
115{
116 // Check BVH desc data
117 if(!bounds.data)
118 return false;
119 if(bounds.stride < sizeof(PxBounds3)) //should be at least one bounds' worth of data
120 return false;
121
122 if(bounds.count == 0)
123 return false;
124
125 if(enlargement<0.0f)
126 return false;
127
128 if(numPrimsPerLeaf>=16)
129 return false;
130
131 return true;
132}
