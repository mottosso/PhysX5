48 struct HullPolygonData
49 {
50 //= ATTENTION! =====================================================================================
51 // Changing the data layout of this class breaks the binary serialization format. See comments for
52 // PX_BINARY_SERIAL_VERSION. If a modification is required, please adjust the getBinaryMetaData
53 // function. If the modification is made on a custom branch, please change PX_BINARY_SERIAL_VERSION
54 // accordingly.
55 //==================================================================================================
56
57 // PT: this structure won't be allocated with PX_NEW because polygons aren't allocated alone (with a dedicated alloc).
58 // Instead they are part of a unique allocation/buffer containing all data for the ConvexHullData class (polygons, followed by
59 // hull vertices, edge data, etc). As a result, ctors for embedded classes like PxPlane won't be called.
60
61 PxPlane mPlane;
62 PxU16 mVRef8;
63 PxU8 mNbVerts;
64 PxU8 mMinIndex;
65
66 PX_FORCE_INLINE PxReal getMin(const PxVec3* PX_RESTRICT hullVertices) const //minimum of projection of the hull along this plane normal
67 {
68 return mPlane.n.dot(hullVertices[mMinIndex]);
69 }
70
71 PX_FORCE_INLINE PxReal getMax() const { return -mPlane.d; } //maximum of projection of the hull along this plane normal
72 };
73
74 PX_FORCE_INLINE void flipData(Gu::HullPolygonData& data)
75 {
76 flip(data.mPlane.n.x);
77 flip(data.mPlane.n.y);
78 flip(data.mPlane.n.z);
79 flip(data.mPlane.d);
80 flip(data.mVRef8);
81 }
82 // PT: if this one breaks, please make sure the 'flipData' function is properly updated.
83 PX_COMPILE_TIME_ASSERT(sizeof(Gu::HullPolygonData) == 20);

167 PX_FORCE_INLINE const PxU8* getFacesByVertices8() const
168 {
169 const char* tmp = reinterpret_cast<const char*>(mPolygons);
170 tmp += sizeof(Gu::HullPolygonData) * mNbPolygons;
171 tmp += sizeof(PxVec3) * mNbHullVertices;
172 tmp += sizeof(PxU8) * mNbEdges * 2;
173 return reinterpret_cast<const PxU8*>(tmp);
174 }
175
176 //If we don't build the convex hull with grb data, we will return NULL pointer
177 PX_FORCE_INLINE const PxU16* getVerticesByEdges16() const
178 {
179 if (mNbEdges.isBitSet())
180 {
181 const char* tmp = reinterpret_cast<const char*>(mPolygons);
182 tmp += sizeof(Gu::HullPolygonData) * mNbPolygons;
183 tmp += sizeof(PxVec3) * mNbHullVertices;
184 tmp += sizeof(PxU8) * mNbEdges * 2;
185 tmp += sizeof(PxU8) * mNbHullVertices * 3;
186 return reinterpret_cast<const PxU16*>(tmp);
187 }
188 return NULL;
189 }
190
191 PX_FORCE_INLINE const PxU8* getVertexData8() const
192 {
193 const char* tmp = reinterpret_cast<const char*>(mPolygons);
194 tmp += sizeof(Gu::HullPolygonData) * mNbPolygons;
195 tmp += sizeof(PxVec3) * mNbHullVertices;
196 tmp += sizeof(PxU8) * mNbEdges * 2;
197 tmp += sizeof(PxU8) * mNbHullVertices * 3;
198 if (mNbEdges.isBitSet())
199 tmp += sizeof(PxU16) * mNbEdges * 2;
200 return reinterpret_cast<const PxU8*>(tmp);
201 }
