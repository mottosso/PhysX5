453// Ensure that the application hasn't tweaked the pack value to less than 8, which would break
454// matching between the API headers and the binaries
455// This assert works on win32/win64, but may need further specialization on other platforms.
456// Some GCC compilers need the compiler flag -malign-double to be set.
457// Apparently the apple-clang-llvm compiler doesn't support malign-double.
458#if PX_APPLE_FAMILY || (PX_CLANG && !PX_ARM)
459 struct PxPackValidation
460 {
461 char _;
462 long a;
463 };
464#elif PX_CLANG && PX_ARM
465 struct PxPackValidation
466 {
467 char _;
468 double a;
469 };
470#else
471 struct PxPackValidation
472 {
473 char _;
474 long long a;
475 };
476#endif
477// clang (as of version 3.9) cannot align doubles on 8 byte boundary when compiling for Intel 32 bit target
478#if !PX_APPLE_FAMILY && !PX_EMSCRIPTEN && !(PX_CLANG && PX_X86)
479 PX_COMPILE_TIME_ASSERT(PX_OFFSET_OF(PxPackValidation, a) == 8);
480#endif
