88 PxU32 tabCount() { return mTags.size() + mInitialTagDepth; }

90 void writeTabs( PxU32 inSize )
91 {
92 inSize += mInitialTagDepth;
93 for ( PxU32 idx =0; idx < inSize; ++idx )
94 mStream << "\t";
95 }
96 void beginTag( const char* inTagname )
97 {
98 closeTag();
99 writeTabs(mTags.size());
100 mTags.pushBack( inTagname );
101 mStream << "<" << inTagname;
102 mTagOpen = true;
103 }
104 void addAttribute( const char* inName, const char* inValue )
105 {
106 PX_ASSERT( mTagOpen );
107 mStream << " " << inName << "=" << "\"" << inValue << "\"";
108 }
109 void closeTag(bool useNewline = true)
110 {
111 if ( mTagOpen )
112 {
113 mStream << " " << ">";
114 if (useNewline )
115 mStream << "\n";
116 }
117 mTagOpen = false;
118 }
119 void doEndOpenTag()
120 {
121 mStream << "</" << mTags.back() << ">" << "\n";
122 }
123 void endTag()
124 {
125 PX_ASSERT( mTags.size() );
126 if ( mTagOpen )
127 mStream << " " << "/>" << "\n";
128 else
129 {
130 writeTabs(mTags.size()-1);
131 doEndOpenTag();
132 }
133 mTagOpen = false;
134 mTags.popBack();
135 }
136 void addContent( const char* inContent )
137 {
138 closeTag(false);
139 mStream << inContent;
140 }
141 void writeContentTag( const char* inTag, const char* inContent )
142 {
143 beginTag( inTag );
144 addContent( inContent );
145 doEndOpenTag();
146 mTags.popBack();
147 }
148 void insertXml( const char* inXml )
149 {
150 closeTag();
151 mStream << inXml;
152 }
