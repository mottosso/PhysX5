222 template<typename TProfileEventType>
223 PX_FORCE_INLINE void doAddProfileEvent(uint16_t eventId, const TProfileEventType& inType)
224 {
225 TScopedLockType lock(TBaseType::mBufferMutex);
226 if (mEventContextInformation == inType.mContextInformation)
227 doAddEvent(static_cast<uint8_t>(inType.getRelativeEventType()), eventId, inType.getRelativeEvent());
228 else
229 {
230 mEventContextInformation = inType.mContextInformation;
231 doAddEvent( static_cast<uint8_t>( getEventType<TProfileEventType>() ), eventId, inType );
232 }
233 }

235 template<typename TDataType>
236 PX_FORCE_INLINE void doAddEvent(uint8_t inEventType, uint16_t eventId, const TDataType& inType)
237 {
238 EventHeader theHeader( inEventType, eventId );
239 //set the header relative timestamp;
240 TDataType& theType( const_cast<TDataType&>( inType ) );
241 uint64_t currentTs = inType.getTimestamp();
242 theType.setupHeader(theHeader, mLastTimestamp);
243 mLastTimestamp = currentTs;
244 sendEvent( theHeader, theType );
245 }
246
247 template<typename TDataType>
248 PX_FORCE_INLINE void sendEvent( EventHeader& inHeader, TDataType& inType )
249 {
250 uint32_t sizeToWrite = sizeof(inHeader) + inType.getEventSize(inHeader);
251 PX_UNUSED(sizeToWrite);
252
253 uint32_t writtenSize = inHeader.streamify( TBaseType::mSerializer );
254 writtenSize += inType.streamify(TBaseType::mSerializer, inHeader);
255
256 PX_ASSERT(writtenSize == sizeToWrite);
257
258 if ( TBaseType::mDataArray.size() >= TBaseType::mBufferFullAmount )
259 flushProfileEvents();
260
261 }
