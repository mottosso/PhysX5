95 enum InternalFlags
96 {
97 //BF_DISABLE_GRAVITY = 1 << 0, // Don't apply the scene's gravity
98
99 BF_HAS_STATIC_TOUCH = 1 << 1, // Set when a body is part of an island with static contacts. Needed to be able to recalculate adaptive force if this changes
100 BF_KINEMATIC_MOVED = 1 << 2, // Set when the kinematic was moved
101
102 BF_ON_DEATHROW = 1 << 3, // Set when the body is destroyed
103
104 BF_IS_IN_SLEEP_LIST = 1 << 4, // Set when the body is added to the list of bodies which were put to sleep
105 BF_IS_IN_WAKEUP_LIST = 1 << 5, // Set when the body is added to the list of bodies which were woken up
106 BF_SLEEP_NOTIFY = 1 << 6, // A sleep notification should be sent for this body (and not a wakeup event, even if the body is part of the woken list as well)
107 BF_WAKEUP_NOTIFY = 1 << 7, // A wake up notification should be sent for this body (and not a sleep event, even if the body is part of the sleep list as well)
108
109 BF_HAS_CONSTRAINTS = 1 << 8, // Set if the body has one or more constraints
110 BF_KINEMATIC_SETTLING = 1 << 9, // Set when the body was moved kinematically last frame
111 BF_KINEMATIC_SETTLING_2 = 1 << 10,
112 BF_KINEMATIC_MOVE_FLAGS = BF_KINEMATIC_MOVED | BF_KINEMATIC_SETTLING | BF_KINEMATIC_SETTLING_2, //Used to clear kinematic masks in 1 call
113 BF_KINEMATIC_SURFACE_VELOCITY = 1 << 11, //Set when the application calls setKinematicVelocity. Actor remains awake until application calls clearKinematicVelocity.
114 BF_IS_COMPOUND_RIGID = 1 << 12, // Set when the body is a compound actor, we dont want to set the sq bounds
115
116 // PT: WARNING: flags stored on 16-bits now.
117 };
