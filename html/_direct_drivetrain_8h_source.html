73class DirectDriveVehicle
74 : public PhysXActorVehicle
75 , public PxVehicleDirectDriveCommandResponseComponent
76 , public PxVehicleDirectDriveActuationStateComponent
77 , public PxVehicleDirectDrivetrainComponent
78{
79public:
80 bool initialize(PxPhysics& physics, const PxCookingParams& params, PxMaterial& defaultMaterial, bool addPhysXBeginEndComponents = true);
81 virtual void destroy();
82
83 virtual void initComponentSequence(bool addPhysXBeginEndComponents);
84
85 void getDataForDirectDriveCommandResponseComponent(
86 const PxVehicleAxleDescription*& axleDescription,
87 PxVehicleSizedArrayData<const PxVehicleBrakeCommandResponseParams>& brakeResponseParams,
88 const PxVehicleDirectDriveThrottleCommandResponseParams*& throttleResponseParams,
89 const PxVehicleSteerCommandResponseParams*& steerResponseParams,
90 PxVehicleSizedArrayData<const PxVehicleAckermannParams>& ackermannParams,
91 const PxVehicleCommandState*& commands, const PxVehicleDirectDriveTransmissionCommandState*& transmissionCommands,
92 const PxVehicleRigidBodyState*& rigidBodyState,
93 PxVehicleArrayData<PxReal>& brakeResponseStates, PxVehicleArrayData<PxReal>& throttleResponseStates,
94 PxVehicleArrayData<PxReal>& steerResponseStates)
95 {
96 axleDescription = &mBaseParams.axleDescription;
97 brakeResponseParams.setDataAndCount(mBaseParams.brakeResponseParams, sizeof(mBaseParams.brakeResponseParams) / sizeof(PxVehicleBrakeCommandResponseParams));
98 throttleResponseParams = &mDirectDriveParams.directDriveThrottleResponseParams;
99 steerResponseParams = &mBaseParams.steerResponseParams;
100 ackermannParams.setDataAndCount(mBaseParams.ackermannParams, sizeof(mBaseParams.ackermannParams)/sizeof(PxVehicleAckermannParams));
101 commands = &mCommandState;
102 transmissionCommands = &mTransmissionCommandState;
103 rigidBodyState = &mBaseState.rigidBodyState;
104 brakeResponseStates.setData(mBaseState.brakeCommandResponseStates);
105 throttleResponseStates.setData(mDirectDriveState.directDriveThrottleResponseStates);
106 steerResponseStates.setData(mBaseState.steerCommandResponseStates);
107 }
108
109 virtual void getDataForDirectDriveActuationStateComponent(
110 const PxVehicleAxleDescription*& axleDescription,
111 PxVehicleArrayData<const PxReal>& brakeResponseStates,
112 PxVehicleArrayData<const PxReal>& throttleResponseStates,
113 PxVehicleArrayData<PxVehicleWheelActuationState>& actuationStates)
114 {
115 axleDescription = &mBaseParams.axleDescription;
116 brakeResponseStates.setData(mBaseState.brakeCommandResponseStates);
117 throttleResponseStates.setData(mDirectDriveState.directDriveThrottleResponseStates);
118 actuationStates.setData(mBaseState.actuationStates);
119 }
120
121 virtual void getDataForDirectDrivetrainComponent(
122 const PxVehicleAxleDescription*& axleDescription,
123 PxVehicleArrayData<const PxReal>& brakeResponseStates,
124 PxVehicleArrayData<const PxReal>& throttleResponseStates,
125 PxVehicleArrayData<const PxVehicleWheelParams>& wheelParams,
126 PxVehicleArrayData<const PxVehicleWheelActuationState>& actuationStates,
127 PxVehicleArrayData<const PxVehicleTireForce>& tireForces,
128 PxVehicleArrayData<PxVehicleWheelRigidBody1dState>& wheelRigidBody1dStates)
129 {
130 axleDescription = &mBaseParams.axleDescription;
131 brakeResponseStates.setData(mBaseState.brakeCommandResponseStates);
132 throttleResponseStates.setData(mDirectDriveState.directDriveThrottleResponseStates);
133 wheelParams.setData(mBaseParams.wheelParams);
134 actuationStates.setData(mBaseState.actuationStates);
135 tireForces.setData(mBaseState.tireForces);
136 wheelRigidBody1dStates.setData(mBaseState.wheelRigidBody1dStates);
137 }
138
139
140 //Parameters and states of the vehicle's direct drivetrain.

142 DirectDrivetrainState mDirectDriveState;
143
144 //The commands that will control the vehicle's transmission
145 PxVehicleDirectDriveTransmissionCommandState mTransmissionCommandState;
146};
