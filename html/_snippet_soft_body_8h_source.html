56 void copyDeformedVerticesFromGPU()
57 {
58 mSoftBody->readData(physx::PxSoftBodyData::ePOSITION_INVMASS, *mPositionInvMass);
59
60 physx::PxTetrahedronMesh* mTetMesh = mSoftBody->getCollisionMesh();
61 physx::PxVec4* positionsInvMass = static_cast<physx::PxVec4*>(mPositionInvMass->map());
62
63 mPositionsInvMass.resize(mTetMesh->getNbVertices());
64 memcpy(mPositionsInvMass.begin(), positionsInvMass, mTetMesh->getNbVertices() * sizeof(physx::PxVec4));
65
66 mPositionInvMass->unmap();
67 }
68
69 physx::PxArray<physx::PxVec4> mPositionsInvMass;
70 physx::PxSoftBody* mSoftBody;
71 physx::PxBuffer* mPositionInvMass;
72};
