164 PX_INLINE PxVec3 computeAABBExtent() const
165 {
166 const PxReal a00 = PxAbs(rot[0][0]);
167 const PxReal a01 = PxAbs(rot[0][1]);
168 const PxReal a02 = PxAbs(rot[0][2]);
169
170 const PxReal a10 = PxAbs(rot[1][0]);
171 const PxReal a11 = PxAbs(rot[1][1]);
172 const PxReal a12 = PxAbs(rot[1][2]);
173
174 const PxReal a20 = PxAbs(rot[2][0]);
175 const PxReal a21 = PxAbs(rot[2][1]);
176 const PxReal a22 = PxAbs(rot[2][2]);
177
178 const PxReal ex = extents.x;
179 const PxReal ey = extents.y;
180 const PxReal ez = extents.z;
181
182 return PxVec3( a00 * ex + a10 * ey + a20 * ez,
183 a01 * ex + a11 * ey + a21 * ez,
184 a02 * ex + a12 * ey + a22 * ez);
185 }
