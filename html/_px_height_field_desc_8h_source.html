163PX_INLINE bool PxHeightFieldDesc::isValid() const
164{
165 if (nbColumns < 2)
166 return false;
167 if (nbRows < 2)
168 return false;
169 if(format != PxHeightFieldFormat::eS16_TM)
170 return false;
171 if (samples.stride < 4)
172 return false;
173 if (convexEdgeThreshold < 0)
174 return false;
175 if ((flags & PxHeightFieldFlag::eNO_BOUNDARY_EDGES) != flags)
176 return false;
177 return true;
178}
